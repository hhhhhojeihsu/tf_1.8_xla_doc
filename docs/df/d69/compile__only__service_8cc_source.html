64 StatusOr<std::vector<std::unique_ptr<AotCompilationResult>>>
65 CompileOnlyService::CompileAheadOfTime(
66  const tensorflow::gtl::ArraySlice<AotComputationInstance> computations,
67  const AotCompilationOptions& options) {
68  std::vector<std::unique_ptr<HloModule>> hlo_modules;
69  for (const AotComputationInstance& instance : computations) {
70  TF_ASSIGN_OR_RETURN(UserComputation * user_computation,
71  computation_tracker_.Resolve(instance.computation));
72  VersionedComputationHandle versioned_handle =
73  user_computation->GetVersionedHandle();
74  const DebugOptions& debug_options = options.debug_options();
75  // Dump computation proto state if flag is set.
76  const string& directory_path = debug_options.xla_dump_computations_to();
77  if (!directory_path.empty()) {
78  TF_ASSIGN_OR_RETURN(
79  std::unique_ptr<SessionModule> session_module,
80  computation_tracker_.SnapshotComputation(versioned_handle.handle));
81  string filename = tensorflow::strings::StrCat(
82  "computation_", versioned_handle.handle.handle(), "__",
83  session_module->entry().name(), "__version_",
84  versioned_handle.version);
85  const string& per_host_path = tensorflow::io::JoinPath(
86  directory_path, tensorflow::port::Hostname());
87  TF_RETURN_IF_ERROR(Executable::DumpToDirectory(per_host_path, filename,
88  *session_module));
89  }
90  TF_ASSIGN_OR_RETURN(
91  std::shared_ptr<const ProgramShape> program_shape,
92  user_computation->ComputeProgramShape(versioned_handle.version));
93  ExecutionOptions execution_options;
94  *execution_options.mutable_debug_options() = debug_options;
95  TF_ASSIGN_OR_RETURN(
96  std::unique_ptr<HloModuleConfig> module_config,
97  CreateModuleConfig(*program_shape, instance.argument_layouts,
98  &execution_options, user_computation));
99  TF_ASSIGN_OR_RETURN(std::unique_ptr<HloModule> hlo_module,
100  computation_tracker_.BuildHloModule(
101  versioned_handle, *module_config,
102  /*include_unreachable_instructions=*/true));
103  TF_RETURN_IF_ERROR(MaybeDumpHloModule(*hlo_module));
104  hlo_modules.push_back(std::move(hlo_module));
105  }
106  return compiler_->CompileAheadOfTime(std::move(hlo_modules), options);
107 }
