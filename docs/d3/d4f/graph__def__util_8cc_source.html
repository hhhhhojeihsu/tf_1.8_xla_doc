65 Status AddDefaultAttrsToGraphDef(GraphDef* graph_def,
66  const OpRegistryInterface& op_registry,
67  int node_offset, bool skip_unknown_ops) {
68  if (node_offset > graph_def->node_size()) {
69  return errors::InvalidArgument(
70  "Tried to add default attrs to GraphDef "
71  "starting at offset ",
72  node_offset, " with total nodes in graph: ", graph_def->node_size());
73  }
74  for (int i = node_offset; i < graph_def->node_size(); ++i) {
75  NodeDef* node_def = graph_def->mutable_node(i);
76  const OpDef* op_def;
77  Status s = op_registry.LookUpOpDef(node_def->op(), &op_def);
78  if (s.ok()) {
79  AddDefaultsToNodeDef(*op_def, node_def);
80  } else if (!skip_unknown_ops) {
81  return s;
82  }
83  }
84  return Status::OK();
85 }
86 static Status RemoveNewDefaultAttrsFromNodeDef(
87  NodeDef* node_def, const OpRegistryInterface& consumer_op_registry,
88  const OpRegistryInterface& producer_op_registry,
89  std::set<std::pair<string, string>>* op_attr_removed) {
90  const OpDef* producer_op_def;
91  const OpDef* consumer_op_def;
92  TF_RETURN_IF_ERROR(
93  producer_op_registry.LookUpOpDef(node_def->op(), &producer_op_def));
94  TF_RETURN_IF_ERROR(
95  consumer_op_registry.LookUpOpDef(node_def->op(), &consumer_op_def));
96  std::vector<string> to_remove;
97  for (const auto& attr : node_def->attr()) {
98  // If the attr is not in consumer_op_def and doesn't start with '_'...
99  if (!str_util::StartsWith(attr.first, "_") &&
100  FindAttr(attr.first, *consumer_op_def) == nullptr) {
101  const OpDef::AttrDef* producer_attr_def =
102  FindAttr(attr.first, *producer_op_def);
103  if (producer_attr_def == nullptr) {
104  return errors::InvalidArgument(
105  "Attr '", attr.first,
106  "' missing in producer's OpDef: ", SummarizeOpDef(*producer_op_def),
107  " but found in node: ", SummarizeNodeDef(*node_def));
108  }
109  // ...and it has the same value as the default in producer,
110  if (producer_attr_def->has_default_value() &&
111  AreAttrValuesEqual(producer_attr_def->default_value(), attr.second)) {
112  // then we will remove it below.
113  to_remove.emplace_back(attr.first);
114  }
115  }
116  }
117  // We separate identifying which attrs should be removed from
118  // actually removing them to avoid invalidating the loop iterators
119  // above.
120  for (const string& attr_name : to_remove) {
121  node_def->mutable_attr()->erase(attr_name);
122  if (op_attr_removed != nullptr) {
123  op_attr_removed->insert(std::make_pair(node_def->op(), attr_name));
124  }
125  }
126  return Status::OK();
127 }
128 static bool IsFunction(const GraphDef& graph_def, const string& op_name) {
129  for (const auto& func_def : graph_def.library().function()) {
130  if (op_name == func_def.signature().name()) return true;
131  }
132  return false;
133 }
134 Status RemoveNewDefaultAttrsFromGraphDef(
135  GraphDef* graph_def, const OpRegistryInterface& consumer_op_registry,
136  const OpRegistryInterface& producer_op_registry,
137  std::set<std::pair<string, string>>* op_attr_removed) {
138  // TODO(joshL): Make IsFunction() faster by collecting the names of
139  // all functions as a preprocessing step.
140  for (int n = 0; n < graph_def->node_size(); ++n) {
141  NodeDef* node_def = graph_def->mutable_node(n);
142  if (!IsFunction(*graph_def, node_def->op())) {
143  TF_RETURN_IF_ERROR(RemoveNewDefaultAttrsFromNodeDef(
144  node_def, consumer_op_registry, producer_op_registry,
145  op_attr_removed));
146  }
147  }
148  for (int f = 0; f < graph_def->library().function_size(); ++f) {
149  FunctionDef* func_def = graph_def->mutable_library()->mutable_function(f);
150  for (int n = 0; n < func_def->node_def_size(); ++n) {
151  NodeDef* node_def = func_def->mutable_node_def(n);
152  if (!IsFunction(*graph_def, node_def->op())) {
153  // TODO(josh11b): Better handling of attrs with placeholder values.
154  TF_RETURN_IF_ERROR(RemoveNewDefaultAttrsFromNodeDef(
155  node_def, consumer_op_registry, producer_op_registry,
156  op_attr_removed));
157  }
158  }
159  }
160  return Status::OK();
161 }
162 void OpsUsedByGraph(const GraphDef& graph_def,
163  std::set<string>* ops_used_in_graph) {
164  // Map function names to definitions.
165  std::unordered_map<string, const FunctionDef*> name_to_function;
166  for (const auto& function : graph_def.library().function()) {
167  name_to_function.insert(
168  std::make_pair(function.signature().name(), &function));
169  }
170  // Collect the sorted list of op names. Since functions can reference
171  // functions, we need a recursive traversal.
172  std::set<string> used_ops; // Includes both primitive ops and functions
173  std::vector<const FunctionDef*> functions_to_process; // A subset of used_ops
174  // Collect the logic to mark an op in a lambda; it'll be used twice below.
175  const auto mark_op_as_used = [&used_ops, &functions_to_process,
176  &name_to_function](const string& op) {
177  if (used_ops.insert(op).second) {
178  // If it's a function, we'll need to process further
179  const auto it = name_to_function.find(op);
180  if (it != name_to_function.end()) {
181  functions_to_process.push_back(it->second);
182  }
183  }
184  };
185  for (const auto& node : graph_def.node()) {
186  mark_op_as_used(node.op());
187  }
188  while (!functions_to_process.empty()) {
189  const FunctionDef* fun = functions_to_process.back();
190  functions_to_process.pop_back();
191  for (const auto& node : fun->node_def()) {
192  mark_op_as_used(node.op());
193  }
194  }
195  // Filter out function names to produce output.
196  // TODO(josh11b): Change the above code to produce this directly.
197  ops_used_in_graph->clear();
198  for (const string& op_name : used_ops) {
199  if (name_to_function.find(op_name) == name_to_function.end()) {
200  ops_used_in_graph->insert(op_name);
201  }
202  }
203 }
204 Status StrippedOpListForGraph(const GraphDef& graph_def,
205  const OpRegistryInterface& op_registry,
206  OpList* stripped_op_list) {
207  std::set<string> used_ops;
208  OpsUsedByGraph(graph_def, &used_ops);
209  // Build the stripped op list in sorted order, ignoring functions.
210  stripped_op_list->clear_op();
211  for (const string& op_name : used_ops) {
212  const OpDef* op_def;
213  TF_RETURN_IF_ERROR(op_registry.LookUpOpDef(op_name, &op_def));
214  OpDef* stripped_op = stripped_op_list->add_op();
215  stripped_op->CopyFrom(*op_def);
216  RemoveDescriptionsFromOpDef(stripped_op);
217  }
218  return Status::OK();
219 }
