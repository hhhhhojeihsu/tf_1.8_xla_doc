298 Status GraphConstructor::EnsureNoNameCollisions() {
299  existing_nodes_.reserve(g_->num_nodes());
300  // Populate existing_nodes_ and existing_prefixes_.
301  for (Node* n : g_->nodes()) {
302  bool already_exists = !existing_nodes_.insert({n->name(), n}).second;
303  if (already_exists) {
304  if (NodeNameInValues(opts_.input_map, n->name())) {
305  return errors::InvalidArgument(
306  "cannot resolve input_map because multiple nodes exist with name '",
307  n->name(), "'");
308  }
309  if (NodeNameInValues(opts_.control_dependencies, n->name())) {
310  return errors::InvalidArgument(
311  "cannot resolve control_dependencies because multiple nodes exist "
312  "with name '",
313  n->name(), "'");
314  }
315  }
316  AddPrefixes(n->name(), &existing_prefixes_);
317  }
318  if (prefix_.empty() && opts_.importing && !opts_.uniquify_names) {
319  for (const NodeDef* n : node_defs_) {
320  const string& name = n->name();
321  if (NameExistsInGraph(name)) {
322  return errors::InvalidArgument("Node name '", name,
323  "' already exists in the Graph");
324  }
325  }
326  } else if (!prefix_.empty()) {
327  StringPiece prefix_no_slash(prefix_);
328  prefix_no_slash.remove_suffix(1);
329  if (!IsValidNodeName(prefix_no_slash, false)) {
330  return errors::InvalidArgument("Imported node name prefix '", prefix_,
331  "' would lead to invalid node names");
332  }
333  if (NameExistsInGraph(prefix_no_slash) && opts_.uniquify_prefix) {
334  prefix_ = strings::StrCat(FindUniqueName(prefix_no_slash), "/");
335  }
336  }
337  return Status::OK();
338 }
339 Status GraphConstructor::ValidateInputMapAndControlDependencies() {
340  for (const auto& mapping : opts_.input_map) {
341  TensorId src = mapping.first;
342  TensorId dst = mapping.second;
343  if (existing_nodes_.count(dst.first) == 0) {
344  return errors::InvalidArgument(
345  "node '", dst.first, "' in input_map does not exist in graph ",
346  "(input_map entry: ", src.ToString(), "->", dst.ToString(), ")");
347  }
348  if ((src.second == Graph::kControlSlot) !=
349  (dst.second == Graph::kControlSlot)) {
350  return errors::InvalidArgument("input_map entry ", src.ToString(), "->",
351  dst.ToString(), " between ",
352  "control edge and non-control edge");
353  }
354  }
355  for (const string& node : opts_.control_dependencies) {
356  if (existing_nodes_.count(node) == 0) {
357  return errors::InvalidArgument(
358  "node '", node,
359  "' in control_dependencies does not exist in "
360  "graph");
361  }
362  }
363  return Status::OK();
364 }
365 Status GraphConstructor::BuildNodeIndex() {
366  // Validate the node names and add them to gdef_nodes_ and gdef_prefixes_.
367  for (int n = 0; n < node_defs_.size(); ++n) {
368  const NodeDef& node_def = *node_defs_[n];
369  if (!IsValidNodeName(node_def.name(), opts_.allow_internal_ops)) {
370  return errors::InvalidArgument(
371  "Node '", node_def.name(),
372  "': Node name contains invalid characters");
373  }
374  if (!gdef_nodes_
375  .insert(std::make_pair(StringPiece(node_def.name()), NodeInfo(n)))
376  .second) {
377  return errors::InvalidArgument("Node '", node_def.name(),
378  "' is not unique");
379  }
380  // Validate the operation's type.
381  if (node_def.op().empty()) {
382  return errors::InvalidArgument("Node '", node_def.name(),
383  "' does not specify an operation");
384  }
385  if (opts_.expect_device_spec && node_def.device().empty()) {
386  return errors::InvalidArgument("Node '", node_def.name(),
387  "' is missing a device specification");
388  }
389  // Validate control edges at end
390  bool in_control_dependence = false;
391  for (int i = 0; i < node_def.input_size(); ++i) {
392  StringPiece input_name = node_def.input(i);
393  if (!input_name.empty() && str_util::StartsWith(input_name, "^")) {
394  in_control_dependence = true;
395  } else if (in_control_dependence) {
396  return errors::InvalidArgument(
397  "Node '", node_def.name(),
398  "': Control dependencies must come after regular dependencies");
399  }
400  }
401  // Update gdef_prefixes_.
402  AddPrefixes(node_def.name(), &gdef_prefixes_);
403  }
404  return Status::OK();
405 }
406 std::unordered_set<string> GetNextIterationNodes(
407  const GraphConstructor::NodeDefSlice& node_defs) {
408  std::unordered_set<string> next_iteration_nodes;
409  for (int n = 0; n < node_defs.size(); ++n) {
410  const NodeDef& node_def = *node_defs[n];
411  if (IsNextIteration(node_def)) {
412  next_iteration_nodes.insert(node_def.name());
413  }
414  }
415  return next_iteration_nodes;
416 }
417 Status GraphConstructor::InitFromEdges() {
418  const int num_nodes = node_defs_.size();
419  pending_count_.reserve(num_nodes);
420  outputs_.resize(num_nodes);
421  std::unordered_set<string> next_iteration_nodes_ =
422  GetNextIterationNodes(node_defs_);
423  // Parse the inputs for each node.
424  for (int n = 0; n < num_nodes; ++n) {
425  const NodeDef& node_def = *node_defs_[n];
426  int pending_count = node_def.input_size();
427  if (IsMerge(node_def)) {
428  // Cycles in the graph are only allowed for while loops. A while loop is
429  // identified by an edge from a NextIteration node to a Merge node. For
430  // such Merge nodes, only wait for one non-control input before
431  // considering the node ready to process in Convert().
432  int32 num_control_edges = 0;
433  bool has_loop_back_edge = false;
434  for (int i = 0; i < node_def.input_size(); ++i) {
435  StringPiece input_name(node_def.input(i));
436  if (str_util::StartsWith(input_name, "^")) {
437  num_control_edges++;
438  } else {
439  TensorId id(ParseTensorName(input_name));
440  if (next_iteration_nodes_.find(id.first.ToString()) !=
441  next_iteration_nodes_.end()) {
442  has_loop_back_edge = true;
443  }
444  }
445  }
446  if (has_loop_back_edge) {
447  pending_count = num_control_edges + 1;
448  }
449  }
450  for (int i = 0; i < node_def.input_size(); ++i) {
451  StringPiece input_name = node_def.input(i);
452  TensorId id(ParseTensorName(input_name));
453  if (opts_.input_map.count(id) == 0) {
454  // If an input is not mapped, then the input should appear in the graph
455  // being imported.
456  auto iter = gdef_nodes_.find(id.first);
457  if (iter == gdef_nodes_.end()) {
458  return errors::InvalidArgument("Node '", node_def.name(),
459  "': Unknown input node '",
460  node_def.input(i), "'");
461  }
462  outputs_[iter->second.gdef_index].push_back(n);
463  } else {
464  // This input is mapped to an existing edge. Therefore this input is
465  // as good as being already processed.
466  --pending_count;
467  DCHECK_GE(pending_count, 0);
468  }
469  }
470  if (pending_count == 0) {
471  ready_.push_back(n);
472  }
473  pending_count_.push_back(pending_count);
474  }
475  return Status::OK();
476 }
477 Status GraphConstructor::ValidateColocationConstraints(
478  const NodeDef& node_def) {
479  if (!opts_.validate_colocation_constraints || !opts_.importing)
480  return Status::OK();
481  const auto iter = node_def.attr().find(kColocationAttrName);
482  if (iter == node_def.attr().end()) return Status::OK();
483  for (const string& c : iter->second.list().s()) {
484  StringPiece s(c);
485  if (str_util::ConsumePrefix(&s, kColocationGroupPrefix) &&
486  gdef_nodes_.find(s) == gdef_nodes_.end()) {
487  return errors::InvalidArgument(
488  "Node '", node_def.name(),
489  "' expects to be colocated with unknown node '", s, "'");
490  }
491  }
492  return Status::OK();
493 }

504 Status GraphConstructor::ValidateShape(Node* node) {
505  if (!opts_.importing || !opts_.validate_shape) return Status::OK();
506  TF_RETURN_IF_ERROR(refiner_->AddNode(node));
507  // For nodes with the _output_shapes attribute, override the shape.
508  std::vector<TensorShapeProto> shape_attrs;
509  const char* kAttrName = "_output_shapes";
510  if (!GetNodeAttr(node->attrs(), kAttrName, &shape_attrs).ok()) {
511  // No _output_shapes attribute, the AddNode call above was sufficient.
512  return Status::OK();
513  }
514  auto* ic = refiner_->GetContext(node);
515  DCHECK(ic != nullptr)
516  << "ShapeRefiner::AddNode() should have created the InferenceContext";
517  if (shape_attrs.size() < node->num_outputs()) {
518  return errors::InvalidArgument(
519  "Node '", node->name(), "' has ", node->num_outputs(),
520  " outputs but the ", kAttrName, " attribute specifies shapes for ",
521  shape_attrs.size(), " outputs");
522  }
523  // NOTE(skyewm): we don't raise an error here because some users depend on
524  // this behavior, even though it's unsafe.
525  // TODO(b/74619486): raise an error.
526  if (shape_attrs.size() > node->num_outputs()) {
527  LOG(WARNING) << "Node '" << node->name() << "' has " << node->num_outputs()
528  << " outputs but the " << kAttrName
529  << " attribute specifies shapes for " << shape_attrs.size()
530  << " outputs. Output shapes may be inaccurate.";
531  }
532  for (int i = 0; i < node->num_outputs(); ++i) {
533  const TensorShapeProto& p = shape_attrs[i];
534  shape_inference::ShapeHandle h;
535  Status s = ic->MakeShapeFromShapeProto(p, &h);
536  if (!s.ok()) {
537  return errors::InvalidArgument("Node '", node->name(), " has an invalid ",
538  kAttrName, " attribute (shape #", i,
539  " error:'", s.error_message(), "'");
540  }
541  s = refiner_->SetShape(node, i, h);
542  if (!s.ok()) {
543  // If the output shape is incompatible with what is inferred
544  // by the graph for a very specific whitelist of ops, then we
545  // ignore this output shape. This can happen if there is a
546  // bug in the shape function for some operation, and the
547  // serialized graph def has the incorrect shape set when
548  // running on a newer binary with the fixed shape function.
549  // This is an escape hatch that allows us to correct shape
550  // functions that are not critical to correct execution but
551  // would cause graphs to fail if imported after correcting.
552  //

609 void RemoveInputs(const std::vector<int>& inputs_to_remove, NodeDef* node_def,
610  std::vector<bool>* input_already_exists) {
611  // Remove 'inputs_to_remove' from 'node_def'
612  // TODO(skyewm): is there a better way to do this?
613  std::vector<string> inputs;
614  inputs.reserve(node_def->input_size());
615  for (int i = 0; i < node_def->input_size(); ++i) {
616  inputs.push_back(node_def->input(i));
617  }
618  node_def->clear_input();
619  for (int i = 0, j = 0; i < inputs.size(); ++i) {
620  if (j < inputs_to_remove.size() && i == inputs_to_remove[j]) {
621  ++j;
622  } else {
623  node_def->add_input(inputs[i]);
624  }
625  }
626  // Remove 'inputs_to_remove' from 'input_already_exists'
627  for (int idx : inputs_to_remove) {
628  input_already_exists->erase(input_already_exists->begin() + idx);
629  }
630  DCHECK_EQ(input_already_exists->size(), node_def->input_size());
631 }
632 void GraphConstructor::RemapNodeDefInputs(
633  NodeDef* node_def, std::vector<bool>* input_already_exists) {
634  DCHECK_EQ(input_already_exists->size(), node_def->input_size());
635  std::set<TensorId> control_inputs;
636  std::vector<int> inputs_to_remove;
637  for (int i = 0; i < node_def->input_size(); ++i) {
638  auto iter = opts_.input_map.find(ParseTensorName(node_def->input(i)));
639  if (iter == opts_.input_map.end()) continue;
640  used_input_map_keys_.insert(iter->first);
641  TensorId new_input = iter->second;
642  if (new_input.second == Graph::kControlSlot) {
643  // Check if we've already remapped a different input to new_input, and if
644  // so remove this input.
645  if (control_inputs.count(new_input) > 0) {
646  inputs_to_remove.push_back(i);
647  continue;
648  }
649  control_inputs.insert(new_input);
650  }
651  node_def->set_input(i, new_input.ToString());
652  (*input_already_exists)[i] = true;
653  }
654  if (!inputs_to_remove.empty()) {
655  RemoveInputs(inputs_to_remove, node_def, input_already_exists);
656  }
657 }
658 void GraphConstructor::AddControlDependencies(
659  NodeDef* node_def, std::vector<bool>* input_already_exists) {
660  // To avoid adding redundant control dependencies to every imported node, skip
661  // nodes that will inherit the dependencies from another imported node.
662  bool inherits_deps = false;
663  for (int i = 0; i < node_def->input_size(); ++i) {
664  // Assume we won't inherit dependencies from remapped inputs that already
665  // exist in the graph. Even if we're wrong, we'll only add redundant
666  // dependencies.
667  if ((*input_already_exists)[i]) continue;
668  // If this input is a backedge, assume we won't inherit the dependencies.
669  // TODO(skyewm): we have many redundant ParseTensorName calls. It could be
670  // worth optimizing these.
671  TensorId id(ParseTensorName(node_def->input(i)));
672  auto iter = gdef_nodes_.find(id.first);
673  DCHECK(iter != gdef_nodes_.end()) << id.first;
674  if (iter->second.node == nullptr) {
675  // Input hasn't been created yet, indicating it's a backedge.
676  continue;
677  }
678  inherits_deps = true;
679  }
680  if (inherits_deps) return;
681  // node_def either has no inputs or all remapped inputs, add the control
682  // dependencies
683  for (const string& control_dep : opts_.control_dependencies) {
684  string input = TensorId(control_dep, Graph::kControlSlot).ToString();
685  const protobuf::RepeatedPtrField<string>& inputs = node_def->input();
686  if (std::find(inputs.begin(), inputs.end(), input) != inputs.end()) {
687  // Control dependency already exists
688  continue;
689  }
690  node_def->add_input(input);
691  input_already_exists->push_back(true);
692  }
693 }
694 void GraphConstructor::AddPrefixToNodeDef(
695  const std::vector<bool>& input_already_exists, NodeDef* node_def) {
696  if (prefix_.empty()) return;
697  node_def->set_name(strings::StrCat(prefix_, node_def->name()));
698  // Update names of input nodes
699  for (int i = 0; i < node_def->input_size(); ++i) {
700  StringPiece input(node_def->input(i));
701  // Skip remapped inputs (which already exist in g_ and are not being
702  // imported).
703  if (input_already_exists[i]) continue;
704  if (str_util::ConsumePrefix(&input, "^")) {
705  node_def->set_input(i, strings::StrCat("^", prefix_, input));
706  } else {
707  node_def->set_input(i, strings::StrCat(prefix_, input));
708  }
709  }
710  // Update names of colocation groups
711  if (node_def->attr().find(kColocationAttrName) != node_def->attr().end()) {
712  auto* list =
713  node_def->mutable_attr()->at(kColocationAttrName).mutable_list();
714  for (int i = 0; i < list->s_size(); ++i) {
715  StringPiece v(list->s(i));
716  if (str_util::ConsumePrefix(&v, kColocationGroupPrefix)) {
717  list->set_s(i, strings::StrCat(kColocationGroupPrefix, prefix_, v));
718  }
719  }
720  }
721 }
722 void GraphConstructor::UniquifyNames(
723  const std::vector<bool>& input_already_exists, NodeDef* node_def) {
724  if (NameExistsInGraph(node_def->name())) {
725  string old_name = node_def->name();
726  node_def->set_name(FindUniqueName(node_def->name()));
727  uniquified_names_[old_name] = node_def->name();
728  // Note that we don't have to update gdef_nodes_ or gdef_prefixes_ with
729  // `name` because we guarantee the original NodeDef names are unique,
730  // meaning we won't generate this name again.
731  }
732  for (int i = 0; i < node_def->input_size(); ++i) {
733  // Skip remapped inputs (which already exist in g_ and are not being
734  // imported).
735  if (input_already_exists[i]) continue;
736  TensorId id = ParseTensorName(node_def->input(i));
737  // We require that UniquifyNames() is called on all NodeDefs in topological
738  // order. This guarantees that node_def's inputs will already be uniquified
739  // if necessary.
740  auto iter = uniquified_names_.find(id.first.ToString());
741  if (iter == uniquified_names_.end()) continue;
742  id.first = iter->second;
743  node_def->set_input(i, id.ToString());
744  }
745 }
746 void GraphConstructor::UpdateUniquifiedColocationNames() {
747  for (const auto& pair : gdef_nodes_) {
748  Node* node = pair.second.node;
749  if (node == nullptr) continue;
750  std::vector<string> coloc_values;
751  Status status =
752  GetNodeAttr(node->attrs(), kColocationAttrName, &coloc_values);
753  if (!status.ok()) continue;
754  bool updated = false;
755  for (int i = 0; i < coloc_values.size(); ++i) {
756  StringPiece val(coloc_values[i]);
757  if (str_util::ConsumePrefix(&val, kColocationGroupPrefix)) {
758  const auto& name_pair = uniquified_names_.find(val.ToString());
759  if (name_pair == uniquified_names_.end()) continue;
760  updated = true;
761  coloc_values[i] =
762  strings::StrCat(kColocationGroupPrefix, name_pair->second);
763  }
764  }
765  if (updated) {
766  node->AddAttr(kColocationAttrName, coloc_values);
767  }
768  }
769 }

781  string name = original_name.ToString();

783  // Check that any generated names don't collide with imported NodeDefs (as
784  // well as nodes in g_).
785  while (NameExistsInGraph(name) || (count > 0 && NameExistsInGraphDef(name))) {
786  name = strings::StrCat(original_name, "_", ++count);

816 Status GraphConstructor::Convert() {
817  // Import functions before adding nodes, since imported nodes may refer to
818  // functions
819  if (library_) {
820  TF_RETURN_IF_ERROR(g_->AddFunctionLibrary(*library_));
821  }
822  std::vector<InputInfo> inputs;
823  int processed = 0;
824  std::vector<bool> input_already_exists;
825  // Process the NodeDefs in topological order.
826  // (InitFromEdges() sets this up by filling in ready_ with nodes that have no
827  // inputs, pending_counts_ with the number of inputs for each node and
828  // outputs_ with the outputs of each node).
829  while (!ready_.empty()) {
830  int o = ready_.back();
831  ready_.pop_back();
832  ++processed;
833  inputs.clear();
834  bool has_data_back_edge = false;
835  const NodeDef& original_node_def = *node_defs_[o];
836  NodeDef imported_node_def;
837  const NodeDef* node_def;
838  // input_already_exists[i] is true iff the i-th input of the node we're
839  // importing refers to a preexisting node in g_ (i.e. input[i] existed prior
840  // to importing node_defs_). Conversely, input_already_exists[i] is false
841  // iff the input refers to a node in node_defs_.
842  input_already_exists.clear();
843  input_already_exists.resize(original_node_def.input_size(), false);
844  if (opts_.importing) {
845  if (opts_.skip_mapped_nodes) {
846  bool is_node_mapped = false;
847  TF_RETURN_IF_ERROR(
848  IsNodeFullyMapped(original_node_def, &is_node_mapped));
849  if (is_node_mapped) {
850  // Skip this node after updating pending_count_ for outputs
851  UpdatePendingCountAndReady(outputs_, o, &pending_count_, &ready_);
852  continue;
853  }
854  }
855  // TODO(ashankar): The line below means an additional copy of the NodeDef,
856  // which can be expensive if the NodeDef contains large tensors in it.
857  // Might make sense to change the API for ImportGraphDef to take a mutable
858  // GraphDef* and avoid the copying.
859  imported_node_def = original_node_def;
860  if (!opts_.input_map.empty()) {
861  // Note that input_already_exists can shrink here
862  RemapNodeDefInputs(&imported_node_def, &input_already_exists);
863  }
864  if (!opts_.control_dependencies.empty()) {
865  // Note that input_already_exists can grow here
866  AddControlDependencies(&imported_node_def, &input_already_exists);
867  }
868  node_def = &imported_node_def;
869  } else {
870  node_def = &original_node_def;
871  }
872  DCHECK_EQ(node_def->input_size(), input_already_exists.size());
873  TF_RETURN_IF_ERROR(ValidateColocationConstraints(*node_def));
874  for (int i = 0; i < node_def->input_size(); ++i) {
875  TensorId id(ParseTensorName(node_def->input(i)));
876  Node* src_node;
877  int src_index;
878  if (!input_already_exists[i]) {
879  // Locate input in newly-imported nodes
880  auto iter = gdef_nodes_.find(id.first);
881  DCHECK(iter != gdef_nodes_.end()) << id.first;
882  src_node = iter->second.node;
883  src_index = id.second;
884  if (src_node == nullptr) has_data_back_edge = true;
885  } else {
886  // Input refers to preexistng node in graph
887  auto iter = existing_nodes_.find(id.first);
888  DCHECK(iter != existing_nodes_.end()) << id.first;
889  src_node = iter->second;
890  src_index = id.second;
891  }
892  if (src_node != nullptr && src_index >= src_node->num_outputs()) {
893  return errors::InvalidArgument(
894  "Node '", node_def->name(), "': Connecting to invalid output ",
895  id.second, " of source node ", id.first, " which has ",
896  src_node->num_outputs(), " outputs");
897  }
898  inputs.push_back(InputInfo(id.first.ToString(), src_node, src_index));
899  }
900  if (has_data_back_edge && !IsMerge(*node_def)) {
901  return errors::InvalidArgument(
902  "Node '", node_def->name(),
903  "' had a back edge, but only Merge nodes can have back edges.");
904  }
905  Node* node;
906  if (opts_.importing) {
907  if (!prefix_.empty()) {
908  AddPrefixToNodeDef(input_already_exists, &imported_node_def);
909  }
910  // Note: no need to uniquify names if the prefix already guarantees
911  // uniqueness
912  if (opts_.uniquify_names && (prefix_.empty() || !opts_.uniquify_prefix)) {
913  UniquifyNames(input_already_exists, &imported_node_def);
914  }
915  TF_RETURN_IF_ERROR(ModifyNodeDefForImport(&imported_node_def));
916  }
917  TF_RETURN_IF_ERROR(MakeNode(*node_def, &node));
918  // Use original_node_def so name StringPiece remains valid
919  gdef_nodes_[original_node_def.name()].node = node;
920  // Add edges from inputs to *node to the graph.
921  for (size_t i = 0; i < inputs.size(); ++i) {
922  if (inputs[i].node == nullptr) {
923  // Record this back edge, which will be added after all nodes
924  // are created.
925  back_edges_.push_back(
926  EdgeInfo(inputs[i].name, inputs[i].index, node, i));
927  } else if (inputs[i].index == Graph::kControlSlot) {
928  g_->AddControlEdge(inputs[i].node, node);
929  } else {
930  TF_RETURN_IF_ERROR(MakeEdge(inputs[i].node, inputs[i].index, node, i));
931  }
932  }
933  // Function shape inference is supported on an opt-in basis per
934  // ShapeRefiner.
935  if (refiner_->function_shape_inference_supported() ||
936  g_->flib_def().Find(node_def->name()) == nullptr) {
937  TF_RETURN_IF_ERROR(ValidateShape(node));
938  }
939  // Update pending_count_ for outputs.
940  UpdatePendingCountAndReady(outputs_, o, &pending_count_, &ready_);
941  }
942  if (processed < node_defs_.size()) {
943  return errors::InvalidArgument(node_defs_.size() - processed,
944  " nodes in a cycle");
945  }
946  return Status::OK();
947 }
948 Status GraphConstructor::AddBackEdges() {
949  // Add the back edges after all nodes are created.
950  for (auto e : back_edges_) {
951  Node* src_node = gdef_nodes_[e.src_name].node;
952  if (e.src_index == Graph::kControlSlot) {
953  g_->AddControlEdge(src_node, e.dst_node);
954  } else {
955  TF_RETURN_IF_ERROR(
956  MakeEdge(src_node, e.src_index, e.dst_node, e.dst_index));
957  }
958  VLOG(2) << "Add back edge: " << src_node->name() << " -> "
959  << e.dst_node->name();
960  }
961  return Status::OK();
962 }
963 Status GraphConstructor::UpdateVersionDef() {
964  if (versions_ == nullptr) return Status::OK();
965  if (!opts_.importing) {
966  g_->set_versions(*versions_);
967  return Status::OK();
968  }
969  VersionDef versions = g_->versions();
970  versions.set_producer(std::min(versions.producer(), versions_->producer()));
971  versions.set_min_consumer(

973  if (versions_->bad_consumers_size() > 0) {

983  g_->set_versions(versions);
984  return Status::OK();

986 Status GraphConstructor::PopulateReturnTensors() {
987  if (opts_.return_tensors.empty()) return Status::OK();
988  for (const TensorId& id : opts_.return_tensors) {
989  auto iter = opts_.input_map.find(id);
990  if (iter == opts_.input_map.end()) {
991  // Locate id in imported nodes
992  auto iter = gdef_nodes_.find(id.first);
993  if (iter == gdef_nodes_.end()) {
994  return errors::InvalidArgument("Requested return tensor '",
995  id.ToString(),
996  "' not found in graph def");
997  }
998  int num_outputs = iter->second.node->num_outputs();
999  if ((id.second < 0 || id.second >= num_outputs) &&
1000  id.second != Graph::kControlSlot) {
1001  return errors::InvalidArgument("Invalid return output ", id.second,
1002  " of node '", id.first, "', which has ",
1003  num_outputs, " output(s)");
1004  }
1005  return_tensors_->push_back({iter->second.node, id.second});
1006  } else {
1007  // id was remapped to existing node
1008  TensorId remapped_id = iter->second;
1009  DCHECK_GT(existing_nodes_.count(remapped_id.first), 0);
1010  Node* node = existing_nodes_[remapped_id.first];
1011  return_tensors_->push_back({node, remapped_id.second});
1012  }
1013  }
1014  return Status::OK();
1015 }

1020  if (iter == gdef_nodes_.end()) {
1021  return errors::InvalidArgument("Requested return node '", name,
1022  "' not found in graph def");

1028 Status GraphConstructor::PopulateMissingUnusedInputMapKeys() {
1029  if (missing_unused_input_map_keys_ == nullptr) return Status::OK();
1030  for (const auto& input_map_pair : opts_.input_map) {
1031  TensorId key = input_map_pair.first;
1032  if (used_input_map_keys_.count(key) > 0) continue;
1033  auto pair = gdef_nodes_.find(key.first);
1034  if (pair == gdef_nodes_.end()) {
1035  // key's node doesn't exist in GraphDef
1036  missing_unused_input_map_keys_->push_back(key);
1037  continue;
1038  }
1039  // Check that key's index is in bounds. Get the number of outputs from the
1040  // NodeDef, rather than the imported Node, since the Node may not exist if
1041  // opts_.skip_mapped_nodes is true.
1042  const NodeDef* node_def = node_defs_[pair->second.gdef_index];
1043  const OpDef* op_def;
1044  TF_RETURN_IF_ERROR(g_->op_registry()->LookUpOpDef(node_def->op(), &op_def));
1045  if (key.second >= op_def->output_arg_size()) {
1046  // key's index out of bounds
1047  missing_unused_input_map_keys_->push_back(key);
1048  }
1049  }
1050  return Status::OK();
1051 }
