40 StatusOr<ComputationHandle> ComputationTracker::LoadSessionModule(
41  const SessionModule& session_module) {
42  tensorflow::mutex_lock lock(computation_mutex_);
43  // For each embedded computation, create a new computation based on its
44  // serialized data, and place the mapping from the old computation handle to
45  // the new computation handle.
46  // Build a mapping from old embedded computation handles to new computation
47  // handles. We build the ID mapping first since the embedded computations are
48  // in no particular order and may refer to each other.
49  std::map<int64, ComputationHandle> old_to_new;
50  for (const SessionComputation& computation :
51  session_module.embedded_computations()) {
52  const int64 old_handle = computation.computation_handle().handle();
53  if (!old_to_new.emplace(old_handle, AllocateHandle()).second) {
54  return InvalidArgument("Duplicate embedded computation handle %lld",
55  old_handle);
56  }
57  }
58  // Create a new computation from each serialized embedded computation.
59  for (const SessionComputation& computation :
60  session_module.embedded_computations()) {
61  const int64 old_handle = computation.computation_handle().handle();
62  const ComputationHandle& new_handle = old_to_new[old_handle];
63  TF_ASSIGN_OR_RETURN(opaque_to_computation_[new_handle.handle()],
64  UserComputation::MakeWithRemapping(
65  computation, new_handle, old_to_new));
66  }
67  // Finally, place the entry computation in the tracker with all of the
68  // remappings populated from the above.
69  const int64 old_handle = session_module.entry().computation_handle().handle();
70  TF_ASSIGN_OR_RETURN(
71  old_to_new[old_handle],
72  LoadSessionComputation(session_module.entry(), &old_to_new));
73  return old_to_new[old_handle];
74 }
81 StatusOr<std::unique_ptr<SessionModule>>
82 ComputationTracker::SnapshotComputation(const ComputationHandle& computation) {
83  TF_ASSIGN_OR_RETURN(UserComputation * user_computation, Resolve(computation));
84  const VersionedComputationHandle entry_versioned_handle =
85  user_computation->GetVersionedHandle();
86  std::set<VersionedComputationHandle> visited;
87  std::list<VersionedComputationHandle> post_order;
88  {
89  tensorflow::mutex_lock lock(computation_mutex_);
90  ComputeComputationPostOrder(entry_versioned_handle, &visited, &post_order);
91  }
92  auto session_module = MakeUnique<SessionModule>();
93  *session_module->mutable_entry() =
94  Resolve(entry_versioned_handle.handle)
95  .ValueOrDie()
96  ->CloneSessionComputation(entry_versioned_handle.version);
97  for (auto it = ++post_order.rbegin(); it != post_order.rend(); ++it) {
98  *session_module->add_embedded_computations() =
99  Resolve(it->handle).ValueOrDie()->CloneSessionComputation(it->version);
100  }
101  return std::move(session_module);
102 }

119 StatusOr<ComputationHandle> ComputationTracker::LoadSessionComputation(
120  const SessionComputation& session_computation,
121  std::map<int64, ComputationHandle>* old_to_new) {
122  TF_RET_CHECK(old_to_new != nullptr);
123  const ComputationHandle new_handle = AllocateHandle();
124  (*old_to_new)[session_computation.computation_handle().handle()] = new_handle;
125  TF_ASSIGN_OR_RETURN(opaque_to_computation_[new_handle.handle()],
126  UserComputation::MakeWithRemapping(
127  session_computation, new_handle, *old_to_new));
128  return new_handle;
129 }
130 StatusOr<UserComputation*> ComputationTracker::ResolveInternal(
131  const ComputationHandle& computation) const {
132  auto it = opaque_to_computation_.find(computation.handle());
133  if (it == opaque_to_computation_.end()) {
134  return NotFound("computation handle not found: %lld", computation.handle());
135  }
136  UserComputation* user_computation = it->second.get();
137  return user_computation;
138 }
139 void ComputationTracker::ComputeComputationPostOrder(
140  const VersionedComputationHandle& versioned_handle,
141  std::set<VersionedComputationHandle>* visited,
142  std::list<VersionedComputationHandle>* post_order) const {
143  if (visited->count(versioned_handle) > 0) {
144  CHECK_EQ(1, visited->count(versioned_handle));
145  return;
146  }
147  UserComputation* computation =
148  ResolveInternal(versioned_handle.handle).ValueOrDie();
149  std::vector<VersionedComputationHandle> embedded_handles =
150  computation->GetEmbeddedComputations(versioned_handle.version);
151  for (const auto& embedded_handle : embedded_handles) {
152  ComputeComputationPostOrder(embedded_handle, visited, post_order);
153  }
154  visited->insert(versioned_handle);
155  post_order->push_back(versioned_handle);
156 }
157 StatusOr<std::unique_ptr<HloModule>> ComputationTracker::BuildHloModule(
158  const VersionedComputationHandle& entry_handle,
159  const HloModuleConfig& config,
160  bool include_unreachable_instructions) const {
161  tensorflow::mutex_lock lock(computation_mutex_);
162  VLOG(1) << "BuildHloModule(" << entry_handle
163  << ", include_unreachable_instructions="
164  << include_unreachable_instructions << ")";
165  XLA_VLOG_LINES(1, ToStringInternal());
166  TF_ASSIGN_OR_RETURN(UserComputation * entry_computation,
167  ResolveInternal(entry_handle.handle));
168  // Build a topological sort of the entry and any embedded computations as a
169  // list. The root of the computation will be the last element in the list.
170  std::set<VersionedComputationHandle> visited;
171  std::list<VersionedComputationHandle> post_order;
172  ComputeComputationPostOrder(entry_handle, &visited, &post_order);
173  // Map from ComputationHandle value and computation version to HloComputation.
174  std::map<VersionedComputationHandle, HloComputation*> hlo_computations;
175  // The resolver lambda resolves VersionedHandles to embedded
176  // HloComputation*. This is required by UserComputation::BuildHloComputation
177  // when lowering calling operations (map, reduce etc).
178  auto resolver = [&hlo_computations](
179  const VersionedComputationHandle& versioned_handle) -> HloComputation* {
180  CHECK_GT(hlo_computations.count(versioned_handle), 0);
181  return hlo_computations.at(versioned_handle);
182  };
183  // Print the post-order list for this entry computation.
184  if (VLOG_IS_ON(2)) {
185  VLOG(2) << "Visiting UserComputations in post order:";
186  for (const VersionedComputationHandle& versioned_handle : post_order) {
187  VLOG(2) << " " << versioned_handle;
188  }
189  }
190  string module_name =
191  tensorflow::strings::StrCat(entry_computation->name(), "_module");
192  auto module = MakeUnique<HloModule>(module_name, entry_handle, config);
193  for (auto versioned_handle : post_order) {
194  UserComputation* computation =
195  ResolveInternal(versioned_handle.handle).ValueOrDie();
196  TF_ASSIGN_OR_RETURN(
197  std::unique_ptr<HloComputation> hlo_computation,
198  computation->BuildHloComputation(versioned_handle.version, resolver,
199  config.debug_options(),
200  include_unreachable_instructions));
201  // Add the newly created computation to VersionedHandle-to-HloComputation
202  // map.
203  DCHECK_EQ(0, hlo_computations.count(versioned_handle));
204  hlo_computations[versioned_handle] = hlo_computation.get();
205  if (computation == entry_computation) {
206  module->AddEntryComputation(std::move(hlo_computation));
207  } else {
208  module->AddEmbeddedComputation(std::move(hlo_computation));
209  }
210  }
211  return std::move(module);
212 }

217 string ComputationTracker::ToStringInternal() const {
218  string out;
219  Appendf(&out, "ComputationTracker(%p):\n", this);
220  for (const auto& handle_computation : opaque_to_computation_) {
221  int64 handle = handle_computation.first;
222  const std::unique_ptr<UserComputation>& computation =
223  handle_computation.second;
224  Appendf(&out, " %4lld : %s \"%s\"\n", handle,
225  computation->GetVersionedHandle().ToString().c_str(),
226  computation->name().c_str());
227  }
228  return out;
229 }
