33 static Status LaunchOpHasKernelForDevice(const DeviceType& device_type) {
34  const OpDef* op_def;
35  TF_RETURN_IF_ERROR(OpRegistry::Global()->LookUpOpDef("_XlaLaunch", &op_def));
36  NodeDef node_def;
37  node_def.set_name("_XlaLaunch-op");
38  node_def.set_op("_XlaLaunch");
39  string kernel_class_name;
40  TF_RETURN_IF_ERROR(FindKernelDef(device_type, node_def, /*KernelDef*/ nullptr,
41  &kernel_class_name));
42  VLOG(1) << "LaunchOpHasKernelForDevice"
43  << " kernel_class_name: " << kernel_class_name;
44  return Status::OK();
45 }

50 /* static */ bool XlaOpRegistry::IsCompatible(const OpRegistration& x,
51  const OpRegistration& y) {
52  if (x.name != y.name) return true;
53  // The registrations refer to the same Op: ensures they are compatible and
54  // are restricted to different device whitelists.
55  if (x.compilation_only != y.compilation_only) {
56  LOG(WARNING) << "Registrations of " << x.name
57  << " have incompatible compilation_only settings.";
58  return false;
59  }
60  if (x.allow_resource_types != y.allow_resource_types) {
61  LOG(WARNING) << "Registrations of " << x.name
62  << " have incompatible allow_resource_types settings.";
63  return false;
64  }
65  if (!x.has_device_whitelist || !y.has_device_whitelist) {
66  LOG(WARNING) << "Registrations of " << x.name
67  << " do not both have device whitelists.";
68  return false;
69  }
70  for (const auto& device : x.device_whitelist) {
71  if (y.device_whitelist.count(device) != 0) {
72  LOG(WARNING) << "Multiple registrations of " << x.name << " on device "
73  << device;
74  return false;
75  }
76  }
77  if (x.compile_time_constant_inputs != y.compile_time_constant_inputs) {
78  LOG(WARNING) << "Registrations of " << x.name
79  << " have incompatible compile time constant inputs.";
80  return false;
81  }
82  return true;
83 }

105 /* static */ bool XlaOpRegistry::GetCompilationDevice(
106  const string& device_name, const DeviceRegistration** registration) {
107  XlaOpRegistry& registry = Instance();
108  // Lazily register the CPU and GPU JIT devices the first time
109  // GetCompilationDevice is called.
110  static void* registration_init = [&registry]() {
111  mutex_lock lock(registry.mutex_);
112  if (LaunchOpHasKernelForDevice(DeviceType(DEVICE_CPU)).ok()) {
113  DeviceRegistration& registration =
114  registry.compilation_devices_[DEVICE_CPU];
115  registration.compilation_device_name = DEVICE_CPU_XLA_JIT;
116  registration.requires_compilation = false;
117  registration.enable_jit_by_default = false;
118  registration.compile_resource_ops = false;
119  }
120  if (LaunchOpHasKernelForDevice(DeviceType(DEVICE_GPU)).ok()) {
121  DeviceRegistration& registration =
122  registry.compilation_devices_[DEVICE_GPU];
123  registration.compilation_device_name = DEVICE_GPU_XLA_JIT;
124  registration.requires_compilation = false;
125  registration.enable_jit_by_default = true;
126  registration.compile_resource_ops = false;
127  }
128  return nullptr;
129  }();
130  (void)registration_init;
131  mutex_lock lock(registry.mutex_);
132  auto it = registry.compilation_devices_.find(device_name);
133  if (it == registry.compilation_devices_.end()) return false;
134  *registration = &it->second;
135  return true;
136 }
140 void XlaOpRegistry::RegisterCompilationKernels() {
141  XlaOpRegistry& registry = Instance();
142  mutex_lock lock(registry.mutex_);
143  if (registry.jit_kernels_registered_) return;
144  registry.jit_kernels_registered_ = true;
145  OpRegistryInterface* op_registry = OpRegistry::Global();
146  for (const auto& op : registry.ops_) {
147  const string& op_name = op.first;
148  const std::unique_ptr<OpRegistration>& op_registration = op.second;
149  const OpDef* op_def;
150  Status lookup_status = op_registry->LookUpOpDef(op_name, &op_def);
151  if (!lookup_status.ok()) {
152  LOG(ERROR) << lookup_status.error_message();
153  XLA_LOG_LINES(
154  ERROR, "Ops registered: \n" +
155  dynamic_cast<OpRegistry*>(op_registry)->DebugString(true));
156  }
157  TF_CHECK_OK(lookup_status);
158  std::unordered_set<string> type_attrs;
159  for (const OpDef::AttrDef& attr_def : op_def->attr()) {
160  if (attr_def.type() == "type" || attr_def.type() == "list(type)") {
161  type_attrs.insert(attr_def.name());
162  }
163  }
164  // Checks there are no type constraints referring to unknown attributes.
165  for (const auto& constraint : op_registration->type_constraints) {
166  if (type_attrs.find(constraint.first) == type_attrs.end()) {
167  LOG(FATAL) << "Unknown type attribute " << constraint.first
168  << " in XLA op registration for " << op_name;
169  }
170  }
171  for (auto& backend : registry.backends_) {
172  // If the operator has a device whitelist, only register on whitelisted
173  // devices.
174  if (op_registration->has_device_whitelist &&
175  op_registration->device_whitelist.find(backend.first) ==
176  op_registration->device_whitelist.end()) {
177  continue;
178  }
179  std::unique_ptr<KernelDef> kdef(new KernelDef);
180  kdef->set_op(op_registration->name);
181  kdef->set_device_type(backend.first);
182  // Constrain each type attribute to the intersection of:
183  // a) the types supported by the backend, and
184  // b) the types allowed by the OpDef, and
185  // c) the type constraints.
186  for (const string& type_attr : type_attrs) {
187  KernelDef::AttrConstraint* attr_constraint = kdef->add_constraint();
188  attr_constraint->set_name(type_attr);
189  auto* allowed_values =
190  attr_constraint->mutable_allowed_values()->mutable_list();
191  const OpDef::AttrDef& op_def_attr = *FindAttr(type_attr, *op_def);
192  const auto* op_def_allowed_types =
193  op_def_attr.has_allowed_values()
194  ? &op_def_attr.allowed_values().list().type()
195  : nullptr;
196  auto constraint_it = op_registration->type_constraints.find(type_attr);
197  const std::set<DataType>* type_constraints =
198  constraint_it != op_registration->type_constraints.end()
199  ? &constraint_it->second
200  : nullptr;
201  for (DataType dtype : backend.second.supported_types) {
202  // Filter out types that aren't allowed by the OpDef.
203  if (op_def_allowed_types != nullptr &&
204  std::find(op_def_allowed_types->begin(),
205  op_def_allowed_types->end(),
206  dtype) == op_def_allowed_types->end()) {
207  continue;
208  }
209  // Filter out types based on the type constraints.
210  if (type_constraints != nullptr &&
211  type_constraints->find(dtype) == type_constraints->end()) {
212  continue;
213  }
214  // Passed all the filters, this type is allowed.
215  allowed_values->add_type(dtype);
216  }
217  if (op_registration->allow_resource_types) {
218  allowed_values->add_type(DT_RESOURCE);
219  }
220  }
221  if (backend.second.op_filter != nullptr &&
222  !backend.second.op_filter(kdef.get())) {
223  continue;
224  }
225  VLOG(2) << "XLA op registration: device: " << backend.first
226  << " op: " << op_name;
227  registry.kernel_registrars_.emplace_back(
228  new kernel_factory::OpKernelRegistrar(
229  new KernelDef(*kdef), "XlaJitOp", op_registration->factory));
230  backend.second.kernel_defs.push_back(std::move(kdef));
231  }
232  }
233 }
234 std::vector<const KernelDef*> XlaOpRegistry::DeviceKernels(
235  const string& compilation_device_name,
236  bool include_compilation_only_kernels) {
237  // Ensure compilation kernels registered.
238  RegisterCompilationKernels();
239  std::vector<const KernelDef*> kernels;
240  XlaOpRegistry& registry = Instance();
241  mutex_lock lock(registry.mutex_);
242  auto it = registry.backends_.find(compilation_device_name);
243  CHECK(it != registry.backends_.end())
244  << "Unknown backend " << compilation_device_name;
245  for (const std::unique_ptr<KernelDef>& k : it->second.kernel_defs) {
246  auto op_iter = registry.ops_.find(k->op());
247  CHECK(op_iter != registry.ops_.end());
248  // The test in IsCompatible ensures that if there are multiple matching
249  // registrations for this op name, they all have the same value of
250  // compilation_only, so only the first match needs to be tested.
251  if (include_compilation_only_kernels ||
252  !op_iter->second->compilation_only) {
253  kernels.push_back(k.get());
254  }
255  }
256  return kernels;
257 }

341 XlaOpRegistrar::XlaOpRegistrar(
342  std::unique_ptr<XlaOpRegistry::OpRegistration> registration) {
343  XlaOpRegistry& registry = XlaOpRegistry::Instance();
344  mutex_lock lock(registry.mutex_);
345  auto existing_ops = registry.ops_.equal_range(registration->name);
346  for (auto existing = existing_ops.first; existing != existing_ops.second;
347  ++existing) {
348  if (!XlaOpRegistry::IsCompatible(*existing->second, *registration)) {
349  LOG(FATAL)
350  << "XLA op registration " << registration->name
351  << " is incompatible with existing registration of the same name.";
352  }
353  }
354  registry.ops_.emplace(registration->name, std::move(registration));
355 }
