45 Status CompileXla(xla::CompileOnlyClient* client,
46  const xla::Computation& computation,
47  const xla::cpu::CpuAotCompilationOptions& aot_opts,
48  CompileResult* compile_result) {
49  // Retrieves arg and result layouts from the computation.
50  // TODO(toddw): Should we let the user choose the major/minor ordering?
51  xla::StatusOr<std::unique_ptr<xla::ProgramShape>> pshape_or =
52  client->GetComputationShape(computation);
53  if (!pshape_or.ok()) {
54  return errors::Unknown("Couldn't get XLA program shape: ",
55  pshape_or.status().error_message());
56  }
57  compile_result->program_shape = *pshape_or.ValueOrDie();
58  xla::ProgramShape* pshape = &compile_result->program_shape;
59  std::vector<const xla::Shape*> arg_layouts;
60  arg_layouts.reserve(pshape->parameters_size());
61  for (int i = 0; i < pshape->parameters_size(); ++i) {
62  arg_layouts.push_back(pshape->mutable_parameters(i));
63  }
64  xla::CompileOnlyClient::AotComputationInstance instance;
65  instance.computation = &computation;
66  instance.argument_layouts = std::move(arg_layouts);
67  instance.result_layout = &pshape->result();
68  xla::StatusOr<std::vector<std::unique_ptr<xla::AotCompilationResult>>>
69  aot_or = client->CompileAheadOfTime({instance}, aot_opts);
70  if (!aot_or.ok()) {
71  return errors::Unknown("XLA compilation failed: ",
72  aot_or.status().error_message());
73  }
74  compile_result->aot =
75  xla::unique_ptr_static_cast<xla::cpu::CpuAotCompilationResult>(
76  std::move(aot_or.ValueOrDie().back()));
77  compile_result->entry_point = aot_opts.entry_point_name();
78  compile_result->pointer_size =
79  xla::CompileOnlyClient::PointerSizeForTriple(aot_opts.triple());
80  return Status::OK();
81 }

91 Status CompileGraph(const GraphDef& graph_def, const tf2xla::Config& config,
92  const MainFlags& flags, CompileResult* compile_result) {
93  // Converts the graph into an XLA computation, and compiles the
94  // computation.
95  // TODO(toddw): Should we let the user pick the XLA cpu vs. gpu client?
96  namespace gpu = perftools::gputools;
97  gpu::Platform* cpu_platform =
98  gpu::MultiPlatformManager::PlatformWithName("Host").ValueOrDie();
99  xla::CompileOnlyClient* client =
100  xla::ClientLibrary::GetOrCreateCompileOnlyClient(cpu_platform)
101  .ValueOrDie();
102  xla::Computation computation;
103  TF_RETURN_IF_ERROR(
104  ConvertGraphDefToXla(graph_def, config, client, &computation));
105  if (!flags.out_session_module.empty()) {
106  TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::SessionModule> module,
107  computation.Snapshot());
108  // Serialize the SessionModule deterministically so that all the outputs of
109  // a tf_library genrule are deterministic.
110  string proto;
111  TF_RET_CHECK(SerializeToStringDeterministic(*module, &proto));
112  TF_RETURN_IF_ERROR(
113  WriteStringToFile(Env::Default(), flags.out_session_module, proto));
114  }
115  xla::cpu::CpuAotCompilationOptions aot_opts(
116  flags.target_triple, flags.target_cpu, flags.target_features,
117  flags.entry_point,
118  xla::cpu::CpuAotCompilationOptions::RelocationModel::BigPic);
119  return CompileXla(client, computation, aot_opts, compile_result);
120 }
