64 Status Main(const MainFlags& flags) {
65  // Process config.
66  tf2xla::Config config;
67  if (flags.config.empty()) {
68  return errors::InvalidArgument("Must specify --config");
69  }
70  TF_RETURN_IF_ERROR(ReadProtoFile(flags.config, &config));
71  TF_RETURN_IF_ERROR(ValidateConfig(config));
72  if (flags.dump_fetch_nodes) {
73  std::set<string> nodes;
74  for (const tf2xla::Fetch& fetch : config.fetch()) {
75  nodes.insert(fetch.id().node_name());
76  }
77  std::cout << str_util::Join(nodes, ",");
78  return Status::OK();
79  }
80  // Read and initialize the graph.
81  if (flags.graph.empty()) {
82  return errors::InvalidArgument("Must specify --graph");
83  }
84  GraphDef graph_def;
85  TF_RETURN_IF_ERROR(ReadProtoFile(flags.graph, &graph_def));
86  CompileResult compile_result;
87  TF_RETURN_IF_ERROR(CompileGraph(graph_def, config, flags, &compile_result));
88  // Write output files.
89  Env* env = Env::Default();
90  const std::vector<char>& obj = compile_result.aot->object_file_data();
91  TF_RETURN_IF_ERROR(WriteStringToFile(env, flags.out_function_object,
92  StringPiece(obj.data(), obj.size())));
93  CodegenOpts codegen_opts;
94  codegen_opts.gen_name_to_index = flags.gen_name_to_index;
95  codegen_opts.gen_program_shape = flags.gen_program_shape;
96  codegen_opts.target_triple = flags.target_triple;
97  if (flags.cpp_class.empty()) {
98  return errors::InvalidArgument("Must specify --cpp_class");
99  }
100  TF_RETURN_IF_ERROR(ParseCppClass(flags.cpp_class, &codegen_opts.class_name,
101  &codegen_opts.namespaces));
102  MetadataResult metadata_result;
103  TF_RETURN_IF_ERROR(
104  GenerateMetadata(codegen_opts, compile_result, &metadata_result));
105  TF_RETURN_IF_ERROR(WriteStringToFile(env, flags.out_metadata_object,
106  metadata_result.object_file_data));
107  string header;
108  TF_RETURN_IF_ERROR(GenerateHeader(codegen_opts, config, compile_result,
109  metadata_result, &header));
110  TF_RETURN_IF_ERROR(WriteStringToFile(env, flags.out_header, header));
111  return Status::OK();
112 }
