206 StatusOr<std::vector<std::vector<const ShapedBuffer*>>>
207 Service::ResolveAndValidateArguments(
208  tensorflow::gtl::ArraySlice<const GlobalDataHandle*> arguments,
209  tensorflow::gtl::ArraySlice<perftools::gputools::StreamExecutor*>
210  stream_executors) {
211  CHECK_EQ(options_.number_of_replicas(), stream_executors.size());
212  std::vector<std::vector<const ShapedBuffer*>> replicated_arguments;
213  replicated_arguments.resize(options_.number_of_replicas());
214  for (size_t i = 0; i < arguments.size(); ++i) {
215  auto buffer_status = allocation_tracker_.Resolve(*arguments[i]);
216  if (!buffer_status.ok()) {
217  return Status(buffer_status.status().code(),
218  StrCat(buffer_status.status().error_message(), ", ",
219  "failed to resolve allocation for parameter ", i));
220  }
221  auto replicated_buffers = buffer_status.ValueOrDie();
222  CHECK_EQ(options_.number_of_replicas(), replicated_buffers.size());
223  for (int replica = 0; replica < options_.number_of_replicas(); ++replica) {
224  const ShapedBuffer* shaped_buffer = replicated_buffers[replica];
225  int replica_device_ordinal = stream_executors[replica]->device_ordinal();
226  // Verify allocation is same platform and device as the execution.
227  if (shaped_buffer->platform() != execute_backend_->platform() ||
228  shaped_buffer->device_ordinal() != replica_device_ordinal) {
229  return InvalidArgument(
230  "argument %lu is on device %s:%d but computation will be executed "
231  "on device %s",
232  i, shaped_buffer->platform()->Name().c_str(),
233  shaped_buffer->device_ordinal(),
234  execute_backend_->device_name(replica_device_ordinal).c_str());
235  }
236  replicated_arguments[replica].push_back(shaped_buffer);
237  }
238  }
239  return replicated_arguments;
240 }
241 StatusOr<std::unique_ptr<HloModuleConfig>> Service::CreateModuleConfig(
242  const ProgramShape& program_shape,
243  tensorflow::gtl::ArraySlice<const Shape*> argument_shapes,
244  const ExecutionOptions* execution_options,
245  const UserComputation* user_computation) {
246  auto config = MakeUnique<HloModuleConfig>(program_shape);
247  auto* computation_layout = config->mutable_entry_computation_layout();
248  if (program_shape.parameters_size() != argument_shapes.size()) {
249  return InvalidArgument("computation takes %d parameters, but %zu given",
250  program_shape.parameters_size(),
251  argument_shapes.size());
252  }
253  for (int i = 0; i < argument_shapes.size(); ++i) {
254  // Verify that shape of arguments matches the shape of the arguments in the
255  // ProgramShape.
256  if (!ShapeUtil::Compatible(*argument_shapes[i],
257  program_shape.parameters(i))) {
258  if (user_computation == nullptr) {
259  return InvalidArgument(
260  "Argument does not match shape of computation parameter %d: want "
261  "%s, got %s",
262  i, ShapeUtil::HumanString(program_shape.parameters(i)).c_str(),
263  ShapeUtil::HumanString(*argument_shapes[i]).c_str());
264  }
265  return InvalidParameterArgument(
266  *user_computation->ParameterMetadata(i).value(),
267  "Argument does not match shape of computation parameter %d: want %s, "
268  "got %s",
269  i, ShapeUtil::HumanString(program_shape.parameters(i)).c_str(),
270  ShapeUtil::HumanString(*argument_shapes[i]).c_str());
271  }
272  TF_RETURN_IF_ERROR(
273  computation_layout->mutable_parameter_layout(i)->CopyLayoutFromShape(
274  *argument_shapes[i]));
275  }
276  if (execution_options != nullptr &&
277  execution_options->has_shape_with_output_layout()) {
278  const auto& shape_with_output_layout =
279  execution_options->shape_with_output_layout();
280  TF_RETURN_IF_ERROR(ValidateResultShapeWithLayout(shape_with_output_layout,
281  program_shape.result()));
282  TF_RETURN_IF_ERROR(
283  computation_layout->mutable_result_layout()->CopyLayoutFromShape(
284  shape_with_output_layout));
285  } else {
286  computation_layout->mutable_result_layout()->Clear();
287  }
288  config->set_replica_count(options_.number_of_replicas());
289  if (execution_options != nullptr) {
290  config->set_seed(execution_options->seed());
291  config->set_debug_options(execution_options->debug_options());
292  } else {
293  config->set_debug_options(legacy_flags::GetDebugOptionsFromFlags());
294  }
295  if (execute_backend_ != nullptr &&
296  execute_backend_->eigen_intra_op_thread_pool() != nullptr) {
297  config->set_intra_op_parallelism_threads(
298  execute_backend_->eigen_intra_op_thread_pool()->NumThreads());
299  }
300  return std::move(config);
301 }

314 StatusOr<std::vector<std::unique_ptr<Executable>>> Service::BuildExecutables(
315  std::vector<VersionedComputationHandle> versioned_handles,
316  std::vector<std::unique_ptr<HloModuleConfig>> module_configs,
317  Backend* backend,
318  std::vector<std::vector<perftools::gputools::StreamExecutor*>> executors,
319  DeviceMemoryAllocator* device_allocator) {
320  VLOG(1) << Printf("BuildExecutable on service %p", this);
321  // Dump computation proto state if flag is set.
322  std::vector<std::unique_ptr<SessionModule>> session_modules;
323  for (int64 i = 0; i < versioned_handles.size(); ++i) {
324  const string& directory_path =
325  module_configs[i]->debug_options().xla_dump_computations_to();
326  const string& other_directory_path =
327  module_configs[i]->debug_options().xla_dump_executions_to();
328  if (directory_path.empty() && other_directory_path.empty()) {
329  continue;
330  }
331  TF_ASSIGN_OR_RETURN(
332  std::unique_ptr<SessionModule> session_module,
333  computation_tracker_.SnapshotComputation(versioned_handles[i].handle));
334  if (!directory_path.empty()) {
335  string filename = Printf("computation_%lld__%s__version_%lld",
336  versioned_handles[i].handle.handle(),
337  session_module->entry().name().c_str(),
338  versioned_handles[i].version);
339  TF_RETURN_IF_ERROR(Executable::DumpToDirectory(directory_path, filename,
340  *session_module));
341  session_modules.push_back(std::move(session_module));
342  }
343  }
344  VLOG(1) << "Computation handles:";
345  for (const VersionedComputationHandle& versioned_handle : versioned_handles) {
346  VLOG(1) << versioned_handle;
347  }
348  CHECK_EQ(versioned_handles.size(), module_configs.size());
349  std::vector<std::unique_ptr<HloModule>> modules;
350  for (int64 i = 0; i < versioned_handles.size(); ++i) {
351  const VersionedComputationHandle& versioned_handle = versioned_handles[i];
352  const HloModuleConfig& config = *module_configs[i];
353  TF_ASSIGN_OR_RETURN(auto module,
354  computation_tracker_.BuildHloModule(
355  versioned_handle, config,
356  /*include_unreachable_instructions=*/true));
357  modules.push_back(std::move(module));
358  }
359  TF_ASSIGN_OR_RETURN(
360  std::vector<std::unique_ptr<Executable>> executables,
361  backend->compiler()->Compile(std::move(modules), std::move(executors),
362  device_allocator));
363  for (size_t i = 0; i < versioned_handles.size(); ++i) {
364  if (!module_configs[i]->debug_options().xla_dump_executions_to().empty()) {
365  executables[i]->set_session_module(std::move(session_modules[i]));
366  }
367  }
368  return std::move(executables);
369 }

396 StatusOr<std::unique_ptr<Executable>> Service::BuildExecutable(
397  const VersionedComputationHandle& versioned_handle,
398  std::unique_ptr<HloModuleConfig> module_config, Backend* backend,
399  se::StreamExecutor* executor, DeviceMemoryAllocator* device_allocator) {
400  VLOG(1) << Printf("BuildExecutable on service %p with handle %s", this,
401  versioned_handle.ToString().c_str());
402  // Dump computation proto state if flag is set.
403  std::unique_ptr<SessionModule> session_module;
404  const string& directory_path =
405  module_config->debug_options().xla_dump_computations_to();
406  const string& other_directory_path =
407  module_config->debug_options().xla_dump_executions_to();
408  if (!directory_path.empty() || !other_directory_path.empty()) {
409  TF_ASSIGN_OR_RETURN(
410  session_module,
411  computation_tracker_.SnapshotComputation(versioned_handle.handle));
412  if (!directory_path.empty()) {
413  string filename = Printf("computation_%lld__%s__version_%lld",
414  versioned_handle.handle.handle(),
415  session_module->entry().name().c_str(),
416  versioned_handle.version);
417  TF_RETURN_IF_ERROR(Executable::DumpToDirectory(directory_path, filename,
418  *session_module));
419  }
420  }
421  TF_ASSIGN_OR_RETURN(
422  std::unique_ptr<HloModule> module,
423  computation_tracker_.BuildHloModule(versioned_handle, *module_config,
424  /*include_unreachable_instructions=*/
425  true));
426  TF_RETURN_IF_ERROR(MaybeDumpHloModule(*module));
427  TF_ASSIGN_OR_RETURN(
428  module, backend->compiler()->RunHloPasses(std::move(module), executor,
429  device_allocator));
430  TF_ASSIGN_OR_RETURN(std::unique_ptr<Executable> executable,
431  backend->compiler()->RunBackend(
432  std::move(module), executor, device_allocator));
433  if (!other_directory_path.empty()) {
434  executable->set_session_module(std::move(session_module));
435  }
436  return std::move(executable);
437 }
438 StatusOr<std::shared_ptr<Executable>> Service::BuildAndCacheExecutable(
439  const VersionedComputationHandle& versioned_handle,
440  std::unique_ptr<HloModuleConfig> module_config, Backend* backend,
441  perftools::gputools::StreamExecutor* executor, ExecutionProfile* profile,
442  DeviceMemoryAllocator* device_allocator) {
443  std::shared_ptr<Executable> executable =
444  compilation_cache_.LookUp(versioned_handle, *module_config);
445  if (executable != nullptr) {
446  // Executable found in the computation cache.
447  if (profile != nullptr) {
448  profile->set_compilation_cache_hit(true);
449  }
450  return executable;
451  }
452  uint64 start_micros =
453  // Avoid reading the clock if we don't want timing info
454  (profile != nullptr) ? tensorflow::Env::Default()->NowMicros() : 0;
455  // Take a copy of the module config, as compilation introduces layouts where
456  // layouts were optional before.
457  HloModuleConfig original_module_config = *module_config;
458  TF_ASSIGN_OR_RETURN(
459  std::unique_ptr<Executable> executable_unique_ptr,
460  BuildExecutable(versioned_handle, std::move(module_config), backend,
461  executor, device_allocator));
462  if (profile != nullptr) {
463  uint64 end_micros = tensorflow::Env::Default()->NowMicros();
464  uint64 milliseconds = (end_micros - start_micros) / 1000;
465  profile->set_compilation_cache_hit(false);
466  profile->set_compile_time_ms(milliseconds);
467  }
468  // Insert executable into the cache.
469  return compilation_cache_.Insert(std::move(executable_unique_ptr),
470  original_module_config);
471 }
472 StatusOr<std::vector<GlobalDataHandle>>
473 Service::ExecuteParallelAndRegisterResult(
474  tensorflow::gtl::ArraySlice<Executable*> executables,
475  tensorflow::gtl::ArraySlice<std::vector<std::vector<const ShapedBuffer*>>>
476  arguments,
477  Backend* backend, tensorflow::gtl::ArraySlice<DeviceHandle> device_handles,
478  tensorflow::gtl::ArraySlice<string> result_tags,
479  ExecutionProfile* profile) {
480  // Streams where the computation are launched, so we can wait on the streams
481  // to complete.
482  std::vector<Pool<se::Stream>::SmartPtr> streams;
483  std::vector<std::unique_ptr<perftools::gputools::Timer>> timers;
484  // Global data handles for the computation results, one for each computation.
485  std::vector<GlobalDataHandle> result_handles;
486  // Device ID to stream executor, populated only with devices that are being
487  // profiled.
488  std::map<int64, se::Stream*> index_to_profiled_streams;
489  TF_ASSIGN_OR_RETURN(DeviceAssignment device_assignment,
490  backend->computation_placer()->AssignDevices(
491  options_.number_of_replicas(), executables.size()));
492  for (int64 i = 0; i < executables.size(); i++) {
493  // Stream executors for the replicas of the current computation.
494  TF_ASSIGN_OR_RETURN(auto replicas, Replicas(*backend, device_handles[i]));
495  CHECK_EQ(replicas.size(), arguments[i].size());
496  std::vector<std::unique_ptr<ShapedBuffer>> result_buffers;
497  for (int64 replica = 0; replica < replicas.size(); ++replica) {
498  TF_ASSIGN_OR_RETURN(Pool<se::Stream>::SmartPtr stream,
499  backend->BorrowStream(replicas[replica]));
500  streams.push_back(std::move(stream));
501  if (replica == 0 && profile != nullptr) {
502  timers.emplace_back(
503  new perftools::gputools::Timer(streams.back()->parent()));
504  streams.back()
505  ->InitTimer(timers.back().get())
506  .ThenStartTimer(timers.back().get());
507  CHECK(timers.front() != nullptr);
508  }
509  if (replica == 0 &&
510  executables[i]->module_config().debug_options().xla_hlo_profile() &&
511  executables[i]->hlo_profiling_enabled()) {
512  index_to_profiled_streams[i] = streams.back().get();
513  }
514  // Set up run options.
515  ExecutableRunOptions options;
516  options.set_stream(streams.back().get());
517  options.set_allocator(backend->memory_allocator());
518  options.set_inter_op_thread_pool(backend->inter_op_thread_pool());
519  options.set_intra_op_thread_pool(
520  backend->eigen_intra_op_thread_pool_device());
521  options.set_device_assignment(&device_assignment);
522  ServiceExecutableRunOptions run_options(options,
523  backend->StreamBorrower());
524  // Asynchronously launch the computation.
525  TF_ASSIGN_OR_RETURN(std::unique_ptr<ShapedBuffer> result,
526  executables[i]->ExecuteAsyncOnStream(
527  &run_options, arguments[i][replica]));
528  if (replica == 0 && profile != nullptr) {
529  streams.back()->ThenStopTimer(timers.back().get());
530  }
531  result_buffers.emplace_back(std::move(result));
532  }
533  TF_ASSIGN_OR_RETURN(GlobalDataHandle handle,
534  allocation_tracker_.RegisterReplicatedBuffers(
535  std::move(result_buffers), result_tags[i]));
536  result_handles.push_back(handle);
537  }
538  // Wait for all executions to complete.
539  for (int64 i = 0; i < streams.size(); ++i) {
540  Status block_status = streams[i]->BlockHostUntilDone();
541  if (!block_status.ok()) {
542  return InternalError("failed to complete execution for stream %lld: %s",
543  i, block_status.error_message().c_str());
544  }
545  }
546  // For every stream that had profiling enabled, obtain and debug-dump the HLO
547  // profile.
548  for (auto& index_to_profiled_stream : index_to_profiled_streams) {
549  int64 device = index_to_profiled_stream.first;
550  se::Stream* stream = index_to_profiled_stream.second;
551  Executable* executable = executables[device];
552  const HloModule& module = executable->module();
553  HloExecutionProfile hlo_profile(&executable->hlo_profile_printer_data(),
554  &executable->hlo_profile_index_map());
555  TF_RETURN_IF_ERROR(
556  executable->PopulateExecutionProfile(&hlo_profile, stream->parent()));
557  XLA_LOG_LINES(
558  tensorflow::INFO,
559  hlo_profile.ToString(streams[0]->parent()->GetDeviceDescription()));
560  hlo_graph_dumper::MaybeDumpHloModule(module, "Service::Execute",
561  &hlo_profile);
562  }
563  if (profile != nullptr) {
564  CHECK(!timers.empty());
565  std::vector<uint64> timer_nanoseconds;
566  timer_nanoseconds.reserve(timers.size());
567  for (auto& timer : timers) {
568  timer_nanoseconds.push_back(timer->Nanoseconds());
569  }
570  uint64 nanoseconds =
571  *std::max_element(timer_nanoseconds.begin(), timer_nanoseconds.end());
572  // Merge in run-time profile information from execution_profile on the
573  // zeroth device.
574  profile->MergeFrom(executables[0]->execution_profile());
575  // Overall execution time (in nanoseconds) from the executor timer.
576  profile->set_compute_and_transfer_time_ns(nanoseconds);
577  // TODO(b/28123297): On GPU we end up including transfer time in
578  // the compute time this way. Instead, we should get the correct
579  // value by measuring it. Setting the field here at least lets
580  // benchmarks provide *some* value for GPU computations.
581  //
582  // TODO(b/28447609): The value in compute_and_transfer_time_ns is actually
583  // the compute time without the transfer time, so this way we get the
584  // correct compute time. We should instead have the correct value for
585  // compute_and_transfer_time and set compute_time to the compute time.
586  if (profile->compute_time_ns() == 0) {
587  profile->set_compute_time_ns(profile->compute_and_transfer_time_ns());
588  }
589  }
590  return result_handles;
591 }
592 StatusOr<GlobalDataHandle> Service::ExecuteAndRegisterResult(
593  Executable* executable,
594  const tensorflow::gtl::ArraySlice<std::vector<const ShapedBuffer*>>
595  arguments,
596  Backend* backend, const string& result_tag, ExecutionProfile* profile) {

610  /*computation_count=*/1));

624  }

626  TF_ASSIGN_OR_RETURN(

637  TF_ASSIGN_OR_RETURN(auto results, executable->ExecuteOnStreams(
638  run_options, replicated_arguments));
639  TF_RET_CHECK(!results.empty());
640  return allocation_tracker_.RegisterReplicatedBuffers(std::move(results),
641  result_tag);

688 tensorflow::Status Service::ExecuteParallel(const ExecuteParallelRequest* arg,
689  ExecuteParallelResponse* result) {
690  VLOG(1) << "running execute-parallel request: " << arg->ShortDebugString();
691  std::vector<std::vector<std::vector<const ShapedBuffer*>>> all_arguments;
692  std::vector<std::vector<perftools::gputools::StreamExecutor*>> all_executors;
693  std::vector<VersionedComputationHandle> versioned_handles;
694  std::vector<std::unique_ptr<HloModuleConfig>> module_configs;
695  std::vector<string> computation_names;
696  std::vector<DeviceHandle> device_handles;
697  int num_requested_devices =
698  std::accumulate(arg->requests().begin(), arg->requests().end(), 0,
699  [](int a, const ExecuteRequest& r) -> int {
700  return a + r.execution_options().device_handles_size();
701  });
702  if (num_requested_devices * options_.number_of_replicas() >
703  execute_backend_->device_count()) {
704  return FailedPrecondition(
705  "there are not enough stream executors to execute %d computations",
706  num_requested_devices);
707  }
708  for (int64 i = 0; i < arg->requests_size(); ++i) {
709  // Get the stream executor for the i'th computation. This stream executor
710  // is one of the executors to run the replicated computation.
711  const ExecutionOptions& execution_options =
712  arg->requests(i).execution_options();
713  // Get the executors.
714  TF_ASSIGN_OR_RETURN(auto executors, GetExecutors(execution_options,
715  arg->requests_size(), i));
716  // Resolve the UserComputation object associated with the requested
717  // computation and compute the program shape.
718  const ExecuteRequest& request = arg->requests(i);
719  TF_ASSIGN_OR_RETURN(UserComputation * user_computation,
720  computation_tracker_.Resolve(request.computation()));
721  VersionedComputationHandle versioned_handle =
722  user_computation->GetVersionedHandle();
723  if (user_computation->request_count(versioned_handle.version) == 0) {
724  return InvalidArgument("computations may not be empty");
725  }
726  TF_ASSIGN_OR_RETURN(
727  std::shared_ptr<const ProgramShape> program_shape,
728  user_computation->ComputeProgramShape(versioned_handle.version));
729  // Get the replicated arguments.
730  TF_ASSIGN_OR_RETURN(auto replicated_arguments,
731  GetArguments(execution_options, request.arguments()));
732  // Create an HloModuleConfig object for the computation, given the shape of
733  // the program and the argument allocations. Here, we care only about the
734  // shapes of the arguments, so, it is sufficient to use the arguments of
735  // replica 0.
736  TF_ASSIGN_OR_RETURN(
737  std::unique_ptr<HloModuleConfig> module_config,
738  CreateModuleConfig(*program_shape, replicated_arguments.front(),
739  request.execution_options(), user_computation));
740  VLOG(3) << "ExecuteParallel created HloModuleConfig computation layout: "
741  << module_config->entry_computation_layout().ToString();
742  // Adds to the vectors to build and execute the computations after the loop.
743  all_arguments.push_back(replicated_arguments);
744  all_arguments.insert(all_arguments.end(), executors.size() - 1, {{}});
745  versioned_handles.push_back(versioned_handle);
746  module_configs.push_back(std::move(module_config));
747  computation_names.insert(computation_names.end(), executors.size(),
748  user_computation->name());
749  all_executors.push_back(executors);
750  device_handles.insert(device_handles.end(),
751  execution_options.device_handles().begin(),
752  execution_options.device_handles().end());
753  }
754  // Build the user computations into HloModules and compile to generate the
755  // executables.
756  //
757  // TODO(jlebar): There's currently no way to pass a device allocator to
758  // ExecuteParallel, so we have to pass a null device_allocator below.
759  TF_ASSIGN_OR_RETURN(
760  std::vector<std::unique_ptr<Executable>> executables,
761  BuildExecutables(versioned_handles, std::move(module_configs),
762  execute_backend_.get(), all_executors,
763  /*device_allocator=*/nullptr));
764  std::vector<Executable*> executable_ptrs;
765  executable_ptrs.reserve(executables.size());
766  for (const auto& executable : executables) {
767  executable_ptrs.push_back(executable.get());
768  }
769  // Execute the generated executables in parallel and return the device
770  // handles for each computation's output.
771  ExecutionProfile profile;
772  TF_ASSIGN_OR_RETURN(
773  std::vector<GlobalDataHandle> outputs,
774  ExecuteParallelAndRegisterResult(executable_ptrs, all_arguments,
775  execute_backend_.get(), device_handles,
776  computation_names, &profile));
777  for (const GlobalDataHandle& output : outputs) {
778  ExecuteResponse response;
779  *response.mutable_output() = output;
780  *response.mutable_profile() = profile;
781  *result->add_responses() = response;
782  }
783  VLOG(1) << "successfully completed 'execute-parallel' request";
784  return tensorflow::Status::OK();
785 }

935 tensorflow::Status Service::Execute(const ExecuteRequest* arg,
936  ExecuteResponse* result) {
937  VLOG(1) << "running execute request: " << arg->ShortDebugString();
938  TF_ASSIGN_OR_RETURN(UserComputation * user_computation,
939  computation_tracker_.Resolve(arg->computation()));
940  VersionedComputationHandle versioned_handle =
941  user_computation->GetVersionedHandle();
942  if (user_computation->request_count(versioned_handle.version) == 0) {
943  return InvalidArgument("computations may not be empty");
944  }
945  // If we received multiple device handles, we must partition the module.
946  if (arg->execution_options().device_handles_size() > 1) {
947  return ExecuteOneToN(arg, result);
948  }
949  TF_ASSIGN_OR_RETURN(
950  std::shared_ptr<const ProgramShape> program_shape,
951  user_computation->ComputeProgramShape(versioned_handle.version));
952  TF_ASSIGN_OR_RETURN(auto replicas, Replicas(*execute_backend_,
953  SingleComputationDeviceHandle()));
954  TF_ASSIGN_OR_RETURN(
955  std::vector<std::vector<const ShapedBuffer*>> replicated_arguments,
956  ResolveAndValidateArguments(arg->arguments(), replicas));
957  // Since we care only about the shapes of the arguments, it is sufficient to
958  // use the arguments of replica 0.
959  TF_ASSIGN_OR_RETURN(
960  std::unique_ptr<HloModuleConfig> module_config,
961  CreateModuleConfig(*program_shape, replicated_arguments.front(),
962  arg->execution_options(), user_computation));
963  VLOG(3) << "Execute created HloModuleConfig computation layout: "
964  << module_config->entry_computation_layout().ToString();
965  TF_ASSIGN_OR_RETURN(
966  std::shared_ptr<Executable> executable,
967  BuildAndCacheExecutable(versioned_handle, std::move(module_config),
968  execute_backend_.get(),
969  execute_backend_->default_stream_executor(),
970  result->mutable_profile()));
971  if (executable->dumping()) {
972  executable->session_module()->set_execution_platform(
973  execute_backend_->platform()->Name());
974  TF_RETURN_IF_ERROR(RecordArguments(
975  replicated_arguments.front(),
976  execute_backend_->default_stream_executor(),
977  execute_backend_->transfer_manager(), executable->session_module()));
978  }
979  TF_ASSIGN_OR_RETURN(
980  *result->mutable_output(),
981  ExecuteAndRegisterResult(
982  executable.get(), replicated_arguments, execute_backend_.get(),
983  "result of " + user_computation->name(), result->mutable_profile()));
984  if (executable->dumping()) {
985  TF_ASSIGN_OR_RETURN(
986  const ShapedBuffer* result_buffer,
987  allocation_tracker_.ResolveForReplica(result->output(), 0));
988  TF_RETURN_IF_ERROR(RecordResult(
989  *result_buffer, execute_backend_->default_stream_executor(),
990  execute_backend_->transfer_manager(), executable->session_module()));
991  TF_RETURN_IF_ERROR(executable->DumpSessionModule());
992  }
993  VLOG(1) << "successfully completed 'execute' request";
994  return tensorflow::Status::OK();
995 }

1273 tensorflow::Status Service::ComputeConstant(const ComputeConstantRequest* arg,

1275  TF_ASSIGN_OR_RETURN(UserComputation * user_computation,
1276  computation_tracker_.Resolve(arg->computation()));
1277  VersionedComputationHandle versioned_handle =
1278  user_computation->GetVersionedHandleAtOperation(arg->operand());
1279  if (user_computation->request_count(versioned_handle.version) == 0) {
1280  return InvalidArgument("computations may not be empty");
1281  }
1282  TF_ASSIGN_OR_RETURN(
1283  bool is_constant,
1284  user_computation->IsConstant(arg->operand(), arg->parameters_size()));
1285  if (!is_constant) {
1286  StatusOr<const OperationRequest*> op_request_status =
1287  user_computation->LookUpRequestForErrorReporting(arg->operand());
1288  string op_request_string = "<unknown operation>";
1289  if (op_request_status.ok()) {
1290  op_request_string = op_request_status.ValueOrDie()->ShortDebugString();

1292  return InvalidArgument(

1311  execution_options.mutable_debug_options()->set_xla_enable_fast_math(false);

1314  *execution_options.mutable_shape_with_output_layout() =

1316  Shape shape_with_output_layout(program_shape.result());
1317  if (arg->has_output_layout()) {
1318  TF_RETURN_IF_ERROR(LayoutUtil::ValidateLayoutForShape(
1319  arg->output_layout(), execution_options.shape_with_output_layout()));
1320  *execution_options.mutable_shape_with_output_layout()->mutable_layout() =
1321  arg->output_layout();
1322  }

1324  CreateModuleConfig(program_shape, {}, execution_options,
1325  user_computation));
1326  // Exclude dead parameter instructions for the purpose of computing constants.
1327  TF_ASSIGN_OR_RETURN(
1328  std::unique_ptr<HloModule> module,
1329  computation_tracker_.BuildHloModule(versioned_handle, *module_config,
1330  /*include_unreachable_instructions=*/
1331  false));
1332  std::vector<std::unique_ptr<Literal>> parameters(arg->parameters_size());
1333  for (int64 i = 0; i < arg->parameters_size(); ++i) {
1334  TF_ASSIGN_OR_RETURN(parameters[i],
1335  Literal::CreateFromProto(arg->parameters(i)));
1336  }

1338  TF_ASSIGN_OR_RETURN(
1339  auto result_literal,
1340  evaluator.Evaluate<std::unique_ptr<Literal>>(*module, parameters));
1341  // Since the shape_with_output_layout option in ExecutionOption is
1342  // non-effective to the Evaluator results, explicit relayout here.
1343  if (arg->has_output_layout()) {
1344  result_literal = result_literal->Relayout(arg->output_layout());
1345  }
1346  *result->mutable_literal() = result_literal->ToProto();
1347  return tensorflow::Status::OK();
