50 typedef std::unordered_map<string, Node*> NodeMap;
51 // Each feed id identifies the positional output of some node, which may consist
52 // of multiple edges. AddPlaceholdersForFeeds has already replaced each fed
53 // tensor with a placeholder. For each feed tensor, replaces all edges so they
54 // point from a new _Arg node instead.
55 Status AddArgNodes(Graph* graph, const NodeMap& node_map,
56  const protobuf::RepeatedPtrField<tf2xla::Feed>& feeds,
57  const std::unordered_map<string, string>& feed_remapping) {
58  for (int arg_index = 0; arg_index < feeds.size(); ++arg_index) {
59  const tf2xla::Feed& feed = feeds[arg_index];
60  // All feeds have been replaced by placeholders.
61  const int output_index = 0;
62  const string key = TensorIdToString(feed.id());
63  const auto remap_it = feed_remapping.find(key);
64  auto node_it = node_map.find(remap_it->second);
65  if (node_it == node_map.end()) {
66  // Strip off the aot_feed_#/ prefix.
67  StringPiece name(remap_it->second);
68  const auto index = name.find('/');
69  if (index > 0) name.remove_prefix(index + 1);
70  return errors::InvalidArgument(
71  "Node is fed but not needed for fetching: ", name);
72  }
73  const Node* feed_node = node_it->second;
74  // TODO(toddw): Invoke shape inference in AddPlaceholdersForFeeds and add a
75  // "_shape" attr if we can determine it. That way the graph will be
76  // initialized with whatever shapes we can infer, while the user can still
77  // explicitly specify or override them.
78  Node* arg_node = nullptr;
79  TF_RETURN_IF_ERROR(
80  NodeBuilder(strings::StrCat("_arg_", arg_index), kArgOp)
81  .Attr("T", BaseType(feed_node->output_type(output_index)))
82  .Attr("index", arg_index)
83  .Attr(kFeedIdAttr, TensorIdToString(feed.id()))
84  .Attr(kShapeAttr, TensorShape(feed.shape()))
85  .Attr(kDebugNameAttr, feed.name())
86  .Finalize(graph, &arg_node));
87  // Collects out-edges from the feed node that have a matching edge index;
88  // these will be replaced with edges from the arg node instead.
89  //
90  // We must collect the edges first and process them in a second pass, since
91  // removing the edge from the graph invalidates feed_node->out_edges.
92  std::vector<const Edge*> feed_edges;
93  for (const Edge* edge : feed_node->out_edges()) {
94  if (edge->src_output() == output_index) {
95  feed_edges.push_back(edge);
96  }
97  }
98  for (const Edge* edge : feed_edges) {
99  graph->AddEdge(arg_node, 0, edge->dst(), edge->dst_input());
100  graph->RemoveEdge(edge);
101  }
102  }
103  return Status::OK();
104 }
105 // Each fetch id identifies the positional output of some node. For each fetch
106 // node, adds a new _Retval node instead, and adds the node to `retval_nodes`.
107 Status AddRetvalNodes(Graph* graph, const NodeMap& node_map,
108  const protobuf::RepeatedPtrField<tf2xla::Fetch>& fetches,
109  std::unordered_set<const Node*>* retval_nodes) {
110  for (int ret_index = 0; ret_index < fetches.size(); ++ret_index) {
111  const tf2xla::TensorId& id = fetches[ret_index].id();
112  auto it = node_map.find(id.node_name());
113  if (it == node_map.end()) {
114  return errors::NotFound("Can't find fetch id: ", TensorIdToString(id));
115  }
116  Node* fetch_node = it->second;
117  if (id.output_index() >= fetch_node->num_outputs()) {
118  return errors::InvalidArgument("Invalid fetch id: ", TensorIdToString(id),
119  ", output index should be < ",
120  fetch_node->num_outputs());
121  }
122  // Connects fetch_node -> retval_node.
123  Node* retval_node = nullptr;
124  TF_RETURN_IF_ERROR(
125  NodeBuilder(strings::StrCat("_retval_", ret_index), kRetvalOp)
126  .Input(fetch_node, id.output_index())
127  .Attr("T", BaseType(fetch_node->output_type(id.output_index())))
128  .Attr("index", ret_index)
129  .Attr(kFetchIdAttr, TensorIdToString(id))
130  .Finalize(graph, &retval_node));
131  retval_nodes->insert(retval_node);
132  }
133  return Status::OK();
134 }
147 Status RewriteAndPruneGraph(
148  Graph* graph, const tf2xla::Config& config,
149  const std::unordered_map<string, string>& feed_remapping) {
150  NodeMap node_map;
151  for (Node* n : graph->nodes()) {
152  node_map[n->name()] = n;
153  }
154  TF_RETURN_IF_ERROR(
155  AddArgNodes(graph, node_map, config.feed(), feed_remapping));
156  std::unordered_set<const Node*> retval_nodes;
157  TF_RETURN_IF_ERROR(
158  AddRetvalNodes(graph, node_map, config.fetch(), &retval_nodes));
159  VLOG(2) << "Post rewrite: "
160  << dump_graph::DumpGraphToFile("tf2xla_post_rewrite", *graph);
161  PruneForReverseReachability(graph, retval_nodes);
162  FixupSourceAndSinkEdges(graph);
163  VLOG(2) << "Post prune: "
164  << dump_graph::DumpGraphToFile("tfcompile_post_prune", *graph);
165  // Sanity-check, to make sure the feeds and fetches still exist post-pruning.
166  std::set<string> missing_feeds, missing_fetches;
167  for (const tf2xla::Feed& feed : config.feed()) {
168  missing_feeds.insert(TensorIdToString(feed.id()));
169  }
170  for (const tf2xla::Fetch& fetch : config.fetch()) {
171  missing_fetches.insert(TensorIdToString(fetch.id()));
172  }
173  for (const Node* n : graph->op_nodes()) {
174  if (n->type_string() == kArgOp) {
175  string feed_id;
176  TF_RETURN_IF_ERROR(GetNodeAttr(n->attrs(), kFeedIdAttr, &feed_id));
177  if (missing_feeds.erase(feed_id) == 0) {
178  return errors::Aborted(kArgOp,
179  " node found with unknown feed id: ", feed_id);
180  }
181  } else if (n->type_string() == kRetvalOp) {
182  string fetch_id;
183  TF_RETURN_IF_ERROR(GetNodeAttr(n->attrs(), kFetchIdAttr, &fetch_id));
184  if (missing_fetches.erase(fetch_id) == 0) {
185  return errors::Aborted(kRetvalOp,
186  " node found with unknown fetch id: ", fetch_id);
187  }
188  }
189  }
190  if (!missing_feeds.empty() || !missing_fetches.empty()) {
191  return errors::Aborted(
192  "Post graph-pruning",
193  ", missing feeds: ", str_util::Join(missing_feeds, ", "),
194  ", missing fetches: ", str_util::Join(missing_fetches, ", "));
195  }
196  return Status::OK();
197 }
198 // CollectArgNodes collects _Arg nodes from the graph, and performs basic
199 // sanity-checking to ensure the index and type attributes of each node are
200 // initialized correctly.
201 Status CollectArgNodes(const Graph& graph, std::vector<Node*>* arg_nodes) {
202  std::map<int, Node*> indexed_arg_nodes;
203  for (Node* n : graph.nodes()) {
204  if (n->type_string() == kArgOp) {
205  int index;
206  TF_RETURN_IF_ERROR(GetNodeAttr(n->attrs(), "index", &index));
207  auto insert_result = indexed_arg_nodes.insert({index, n});
208  if (!insert_result.second) {
209  const Node* dup = insert_result.first->second;
210  return errors::InvalidArgument(
211  "Multiple ", kArgOp, " nodes with index ", index, ", ",
212  n->DebugString(), " and ", dup->DebugString());
213  }
214  }
215  }
216  arg_nodes->clear();
217  for (const auto& index_node : indexed_arg_nodes) {
218  if (index_node.first != arg_nodes->size()) {
219  return errors::InvalidArgument("Expected ", kArgOp, " node with index ",
220  arg_nodes->size(), ", but got index ",
221  index_node.first);
222  }
223  arg_nodes->push_back(index_node.second);
224  }
225  return Status::OK();
226 }
227 // Fills in xla_args from the corresponding _Arg nodes in the graph.
228 Status CreateXlaArgs(const Graph& graph,
229  std::vector<XlaCompiler::Argument>* xla_args) {
230  std::vector<Node*> arg_nodes;
231  TF_RETURN_IF_ERROR(CollectArgNodes(graph, &arg_nodes));
232  for (const Node* node : arg_nodes) {
233  XlaCompiler::Argument arg;
234  arg.kind = XlaCompiler::Argument::kParameter;
235  TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), "T", &arg.type));
236  TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), kShapeAttr, &arg.shape));
237  TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), kDebugNameAttr, &arg.name));
238  xla_args->push_back(arg);
239  }
240  return Status::OK();
241 }
253 Status ConvertGraphToXla(std::unique_ptr<Graph> graph, xla::Client* client,
254  xla::Computation* computation) {
255  XlaOpRegistry::RegisterCompilationKernels();
256  for (Node* node : graph->nodes()) {
257  node->set_assigned_device_name(
258  strings::StrCat("/device:", DEVICE_CPU_XLA_JIT));
259  }
260  std::vector<XlaCompiler::Argument> xla_args;
261  TF_RETURN_IF_ERROR(CreateXlaArgs(*graph, &xla_args));
262  // Compile the graph into an XLA computation.
263  XlaCompiler::Options compiler_options;
264  compiler_options.client = client;
265  DeviceType device_type(DEVICE_CPU_XLA_JIT);
266  compiler_options.device_type = &device_type;
267  compiler_options.flib_def = &graph->flib_def();
268  compiler_options.graph_def_version = graph->versions().producer();
269  compiler_options.allow_cpu_custom_calls = true;
270  XlaCompiler compiler(compiler_options);
271  XlaCompiler::CompilationResult result;
272  TF_RETURN_IF_ERROR(compiler.CompileGraph(XlaCompiler::CompileOptions(),
273  "tfcompile", std::move(graph),
274  xla_args, &result));
275  *computation = std::move(*result.computation);
276  int num_const_results = 0;
277  for (int i = 0; i < result.outputs.size(); ++i) {
278  // Ending up with const results (i.e. output args) is an error, since it
279  // means that one or more fetches that the user specified will be dropped
280  // from the generated function. It's most likely a configuration error,
281  // since the user shouldn't be asking for output args that end up as consts.
282  //
283  // TODO(toddw): Provide a way for the user to access const output args,
284  // e.g. perhaps hard-coded into the header, or somehow copied into the
285  // output buffers.
286  if (result.outputs[i].is_constant) {
287  ++num_const_results;
288  LOG(ERROR) << "ConstRetVal index:" << i
289  << " value:" << result.outputs[i].constant_value.DebugString();
290  }
291  }
292  if (num_const_results > 0) {
293  return errors::Unimplemented(
294  "Conversion from TensorFlow graph to XLA resulted in ",
295  num_const_results,
296  " constant results. The configuration of "
297  "the output args (i.e. fetch ids) is probably wrong.");
298  }
299  return Status::OK();
300 }
324 Status InitGraph(const GraphDef& graph_def, const tf2xla::Config& config,
325  std::unique_ptr<Graph>* graph) {
326  TF_RETURN_IF_ERROR(ValidateConfig(config));
327  FunctionLibraryDefinition flib_def(OpRegistry::Global(), graph_def.library());
328  std::unique_ptr<Graph> g(new Graph(flib_def));
329  // Replace references to fed tensors with references to newly added
330  // placeholders.
331  GraphDef first_copy_def = graph_def;
332  // Maps from name:port of a feed to the name:port of the placeholder to use.
333  std::unordered_map<string, string> feed_remapping;
334  TF_RETURN_IF_ERROR(AddPlaceholdersForFeeds(config, g->op_registry(),
335  &feed_remapping, &first_copy_def));
336  // Prune the GraphDef first so that unknown ops that we aren't compiling get
337  // filtered out.
338  GraphDef second_copy_def;
339  TF_RETURN_IF_ERROR(
340  PruneGraphDefInto(config, first_copy_def, &second_copy_def));
341  TF_RETURN_IF_ERROR(AddDefaultAttrsToGraphDef(
342  &second_copy_def, *g->op_registry(), /*node_offset=*/0));
343  TF_RETURN_IF_ERROR(ConvertGraphDefToGraph(GraphConstructorOptions(),
344  second_copy_def, g.get()));
345  TF_RETURN_IF_ERROR(RewriteAndPruneGraph(g.get(), config, feed_remapping));
346  *graph = std::move(g);
347  return Status::OK();
348 }
