52 Status CheckFeedFetchNameConflicts(const string& kind,
53  const std::set<string>& names) {
54  // We don't allow the feeds or fetches to contain both "foo" and "foo_data",
55  // since that will cause a collision in codegen symbols.
56  for (const string& name : names) {
57  const string name_data(name + "_data");
58  if (names.find(name_data) != names.end()) {
59  return errors::InvalidArgument("conflicting ", kind, " name: ", name,
60  " and ", name_data);
61  }
62  }
63  return Status::OK();
64 }

69 Status ValidateConfig(const tf2xla::Config& config) {
70  std::set<string> names;
71  for (const tf2xla::Feed& feed : config.feed()) {
72  TF_RETURN_IF_ERROR(ValidateTensorId(feed.id()));
73  TF_RETURN_IF_ERROR(TensorShape::IsValidShape(feed.shape()));
74  TF_RETURN_IF_ERROR(CheckNameDuplicates("feed", feed.name(), &names));
75  }
76  TF_RETURN_IF_ERROR(CheckFeedFetchNameConflicts("feed", names));
77  names.clear();
78  for (const tf2xla::Fetch& fetch : config.fetch()) {
79  TF_RETURN_IF_ERROR(ValidateTensorId(fetch.id()));
80  TF_RETURN_IF_ERROR(CheckNameDuplicates("fetch", fetch.name(), &names));
81  }
82  TF_RETURN_IF_ERROR(CheckFeedFetchNameConflicts("fetch", names));
83  if (config.fetch().empty()) {
84  return errors::InvalidArgument("fetches must be specified");
85  }
86  return Status::OK();
87 }
97 Status AddPlaceholdersForFeeds(
98  const tf2xla::Config& config, const OpRegistryInterface* op_registry,
99  std::unordered_map<string, string>* feed_remapping, GraphDef* graph_def) {
100  struct PlaceholderInfo {
101  const tf2xla::Feed* feed = nullptr; // point to Feed in <config>.
102  string placeholder_name;
103  DataType data_type = DT_INVALID;
104  };
105  // Put each fed tensor into a map by name:port. A map is used for determinism
106  // when creating placeholders (genrules want deterministic output).
107  std::map<string, PlaceholderInfo> placeholder_info;
108  for (int i = 0; i < config.feed_size(); ++i) {
109  const tf2xla::Feed* feed = &config.feed(i);
110  const string name_port = TensorIdToString(feed->id());
111  PlaceholderInfo& info = placeholder_info[name_port];
112  info.feed = feed;
113  info.placeholder_name = strings::StrCat(
114  "aot_feed_", feed->id().output_index(), "/", feed->id().node_name());
115  (*feed_remapping)[name_port] = info.placeholder_name;
116  }
117  // Verify node exists and determine data type.
118  std::unordered_map<string, const NodeDef*> name_to_node;
119  for (int i = 0; i < graph_def->node_size(); ++i) {
120  name_to_node[graph_def->node(i).name()] = &graph_def->node(i);
121  }
122  for (auto it = placeholder_info.begin(); it != placeholder_info.end(); ++it) {
123  PlaceholderInfo& info = it->second;
124  const tf2xla::TensorId& feed_id = info.feed->id();
125  // Find the existing node and determine data type.
126  auto node_it = name_to_node.find(feed_id.node_name());
127  if (node_it == name_to_node.end()) {
128  return errors::NotFound("Can't find feed node: ",
129  TensorIdToString(feed_id));
130  }
131  const NodeDef* existing = node_it->second;
132  if (info.feed->type() != DT_INVALID) {
133  info.data_type = info.feed->type();
134  } else {
135  // Build the node in order to infer its type.
136  // Must first add default attrs as well, so do this in a copied GraphDef.
137  GraphDef gd;
138  *gd.mutable_versions() = graph_def->versions();
139  *gd.add_node() = *existing;
140  TF_RETURN_IF_ERROR(
141  AddDefaultAttrsToGraphDef(&gd, *op_registry, 0 /*node_offset*/));
142  // Now build the node from the copied node def.
143  Graph g(op_registry);
144  g.set_versions(graph_def->versions());
145  Status status;
146  Node* feed_node = g.AddNode(gd.node(0), &status);
147  TF_RETURN_IF_ERROR(status);
148  if (info.feed->id().output_index() < feed_node->num_outputs()) {
149  info.data_type =
150  BaseType(feed_node->output_type(info.feed->id().output_index()));
151  } else {
152  return errors::InvalidArgument(
153  "Invalid output_index ", info.feed->id().output_index(),
154  " for feed node ", info.feed->id().node_name());
155  }
156  }
157  }
158  // Create placeholders. Note that we could avoid creating a placeholder for
159  // feeds which are already placeholders, but we omit that to avoid more cases
160  // in this code.
161  for (auto it = placeholder_info.begin(); it != placeholder_info.end(); ++it) {
162  const PlaceholderInfo& info = it->second;
163  NodeDef* d = graph_def->add_node();
164  d->set_name(info.placeholder_name);
165  d->set_op("PlaceholderV2");
166  auto& attr_map = *d->mutable_attr();
167  attr_map["dtype"].set_type(info.data_type);
168  *attr_map["shape"].mutable_shape() = info.feed->shape();
169  }
170  // Rewrite references to the fed tensors to refer to the placeholder.
171  for (int i = 0; i < graph_def->node_size(); ++i) {
172  NodeDef* node_def = graph_def->mutable_node(i);
173  for (int j = 0; j < node_def->input_size(); ++j) {
174  auto id = ParseTensorName(node_def->input(j));
175  auto it = placeholder_info.find(id.ToString());
176  if (it != placeholder_info.end()) {
177  node_def->set_input(j, it->second.placeholder_name);
178  }
179  }
180  }
181  return Status::OK();
182 }
188 Status PruneGraphDefInto(const tf2xla::Config& config, const GraphDef& in,
189  GraphDef* out) {
190  *out = in;
191  out->clear_node();
192  // Tensors needed for feeding.
193  std::set<std::pair<string, int>> feed_tensors;
194  for (const tf2xla::Feed& feed : config.feed()) {
195  feed_tensors.insert(
196  std::make_pair(feed.id().node_name(), feed.id().output_index()));
197  }
198  // Maps node name to reachability.
199  std::unordered_map<string, std::pair<bool, const NodeDef*>> node_by_name;
200  for (const NodeDef& node : in.node()) {
201  node_by_name[node.name()] = std::pair<bool, const NodeDef*>(false, &node);
202  }
203  // Traverse.
204  std::queue<string> name_queue;
205  for (int i = 0; i < config.fetch_size(); ++i) {
206  name_queue.push(config.fetch(i).id().node_name());
207  }
208  while (!name_queue.empty()) {
209  const string name = name_queue.front();
210  name_queue.pop();
211  auto find_it = node_by_name.find(name);
212  if (find_it == node_by_name.end()) {
213  return errors::InvalidArgument("While pruning graph, node ", name,
214  " needed but not found in the graph.");
215  }
216  auto& map_entry = find_it->second;
217  if (map_entry.first) {
218  continue;
219  }
220  map_entry.first = true;
221  // Push input nodes of the currently visited node to name_queue.
222  for (const string& in_edge : map_entry.second->input()) {
223  auto id = ParseTensorName(in_edge);
224  const string node_name = id.first.ToString();
225  if (feed_tensors.find(std::make_pair(node_name, id.second)) ==
226  feed_tensors.end()) {
227  name_queue.push(node_name);
228  } else {
229  // The input tensor is from an edge that is being fed. Therefore,
230  // we skip recursing down that edge, to avoid requiring nodes that
231  // may not be needed (note that the input node may still be added
232  // to name_queue later if one of its output edges is not being fed).
233  }
234  }
235  }
236  // Copy over, preserving order of original and only nodes that are reachable
237  // from the fetches.
238  out->mutable_node()->Reserve(in.node_size());
239  for (const NodeDef& node : in.node()) {
240  if (node_by_name[node.name()].first) {
241  *out->add_node() = node;
242  }
243  }
244  return Status::OK();
245 }

249 Status SetNodeShardingFromNeighbors(Node* n, bool out_edges) {
250  int core = -1;
251  const Node* matching_node = nullptr;
252  for (const Edge* edge : (out_edges ? n->out_edges() : n->in_edges())) {
253  if (edge->IsControlEdge()) continue;
254  const Node* possible_match = out_edges ? edge->dst() : edge->src();
255  TF_ASSIGN_OR_RETURN(
256  tensorflow::gtl::optional<xla::OpSharding> sharding,
257  ParseShardingFromDevice(
258  *possible_match,
259  /*num_cores_per_replica=*/std::numeric_limits<int32>::max()));
260  if (sharding.has_value()) {
261  TF_RET_CHECK(sharding.value().type() ==
262  xla::OpSharding::Type::OpSharding_Type_MAXIMAL);
263  const int core_annotation = sharding.value().tile_assignment_devices(0);
264  if (core == -1 || core > core_annotation) {
265  core = core_annotation;
266  matching_node = possible_match;
267  }
268  }
269  }
270  if (matching_node != nullptr) {
271  n->set_assigned_device_name(matching_node->assigned_device_name());
272  n->set_requested_device(matching_node->requested_device());
273  }
274  return Status::OK();
275 }
